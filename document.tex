	% -----------------------------------------------------------------------
	\section*{Introduction}
	
	This note develops a self-contained algebraic framework for power sums
	$s_n = x^n + y^n + z^n$ and their relatives.  The central object is a
	sequence $K(n)$ built from the antisymmetric weights of $x, y, z$; together
	with two companion sequences $L$ and $M$, it forms a basis for all sequences
	satisfying the same linear recurrence as the power sums.
	
	All arguments use only the recurrence and elementary algebra; no
	advanced number theory or analysis is required.
	
	% -----------------------------------------------------------------------
	\section{Setup}
	
	Throughout, $x$, $y$, $z$ are distinct and nonzero complex (or real)
	numbers.  We write their elementary symmetric polynomials as
	\[
	e_1 = x+y+z, \qquad e_2 = xy+yz+zx, \qquad e_3 = xyz,
	\]
	and define the \emph{minimal polynomial}
	\[
	P(t) = t^3 - e_1 t^2 + e_2 t - e_3 = (t-x)(t-y)(t-z).
	\]
	Since $e_3 = xyz \neq 0$ we may divide by any power of $x$, $y$, or $z$,
	and in particular extend all recurrences to all integers $n \in \mathbb{Z}$.
	
	\subsection*{Cyclic sum notation}
	
	Many expressions in this paper are cyclic in $x, y, z$.  We write
	\[
	\cyc{g(x)} \;:=\; g(x) + g(y) + g(z)
	\]
	for the cyclic sum of any expression $g$, where the implicit
	substitution rule is $x \to y \to z \to x$.  For example,
	$e_1 = \cyc{x}$, $e_2 = \cyc{xy}$, $e_3 = xyz$, and more usefully:
	\[
	\cyc{x^n(y-z)} \;=\; x^n(y-z) + y^n(z-x) + z^n(x-y).
	\]
	This notation eliminates the need to write all three terms of any
	cyclic identity.

	\section{Weighted Power Sums}
	
	\begin{definition}
		For any integer $n$ and a weight triple $(k, l, m)$, define
		\[
		f_n(k,l,m) = k x^n + l y^n + m z^n = \cyc{k\, x^n},
		\]
		where in the cyclic notation it is understood that the weights cycle
		as $(k,l,m) \to (l,m,k)$ alongside $(x,y,z) \to (y,z,x)$.
	\end{definition}
	
	Special cases of immediate interest:
	\[
	s_n \;:=\; f_n(1,1,1) \;=\; x^n + y^n + z^n \;=\; \cyc{x^n}.
	\]
	
	The key properties of $f_n$ are linearity in weights and the universal
	recurrence inherited from $P(t)$.
	
	\begin{theorem}[Recurrence]
		\label{thm:recurrence}
		For any fixed weights $(k,l,m)$ and any integer $n$,
		\[
		f_{n+3}(k,l,m) = e_1\, f_{n+2}(k,l,m) - e_2\, f_{n+1}(k,l,m) + e_3\, f_n(k,l,m).
		\]
	\end{theorem}
	\begin{proof}
		Each of $x,y,z$ satisfies $t^3 = e_1 t^2 - e_2 t + e_3$.  Multiply by
		$t^n$, weight by $k, l, m$ respectively, and sum.
	\end{proof}
	
	Because $e_3 \neq 0$ the recurrence can also be run backwards, making
	$f_n(k,l,m)$ well-defined for all $n \in \mathbb{Z}$.
	
	% -----------------------------------------------------------------------
	\section{The Basis $K$, $L$, $M$}
	
	Any sequence satisfying the recurrence of Theorem~\ref{thm:recurrence}
	is determined by its values at three consecutive integers.  We single
	out the three basis sequences with initial values forming the identity
	matrix at $n = 0, 1, 2$.
	
	\subsection{Definition via the antisymmetric weights}
	
	The triple of \emph{difference weights}
	\[
	(w_x,\, w_y,\, w_z) \;:=\; (y-z,\; z-x,\; x-y)
	\]
	satisfies $w_x + w_y + w_z = 0$ and is nonzero since $x,y,z$ are
	distinct.  A short computation gives
	\[
	f_0(w_x,w_y,w_z) = 0, \quad f_1(w_x,w_y,w_z) = 0, \quad
	f_2(w_x,w_y,w_z) = \cyc{x^2(y-z)} =: \Delta,
	\]
	where $\Delta = -(y-z)(z-x)(x-y) \neq 0$.  (The $f_2$ identity follows
	by straightforward algebra, or see Section~\ref{sec:substitutions}.)
	
	\begin{definition}
		\label{def:K}
		For any integer $n$, define
		\[
		K(n) \;=\; \frac{f_n(w_x,\,w_y,\,w_z)}{\Delta}
		\;=\; \frac{\cyc{x^n(y-z)}}{\cyc{x^2(y-z)}}.
		\]
	\end{definition}
	
	\begin{definition}
		\label{def:LM}
		\[
		L(n) = K(n+1) - e_1\,K(n), \qquad M(n) = e_3\,K(n-1).
		\]
	\end{definition}
	
	The initial values follow immediately:
	
	\[
	\renewcommand{\arraystretch}{1.3}
	\begin{array}{c|ccc}
		n & 0 & 1 & 2 \\\hline
		K(n) & 0 & 0 & 1 \\
		L(n) & 0 & 1 & 0 \\
		M(n) & 1 & 0 & 0
	\end{array}
	\]
	
	The triple $(K, L, M)$ forms the \emph{standard basis} for all integer
	sequences satisfying the recurrence: the columns at $n=0,1,2$ are
	exactly the identity matrix.  All three sequences satisfy the same
	recurrence as $f_n$, since $K$ inherits it from $f_n$, and $L$, $M$
	are linear combinations of shifts of $K$.
	
	\begin{remark}
		The explicit values of $K$ at small $n$ are
		$K(2)=1$, $K(3)=e_1$, $K(4)=e_1^2-e_2$, and so on; $L$ and $M$ are
		recovered from $K$ via the definitions above.  The backward extension
		gives $K(-1) = 1/e_3$, consistent with $M(0) = e_3 K(-1) = 1$.
	\end{remark}
	
	% -----------------------------------------------------------------------
	\section{The Decomposition Theorem}
	\label{sec:decomposition}
	
	The following is the central engine of everything that follows.
	
	\begin{theorem}[Decomposition]
		\label{thm:decomposition}
		For any weights $(k,l,m)$ and any integer $n$,
		\[
		f_n(k,l,m) \;=\; K(n)\,f_2(k,l,m) \;+\; L(n)\,f_1(k,l,m) \;+\; M(n)\,f_0(k,l,m).
		\]
	\end{theorem}
	\begin{proof}
		Both sides satisfy the same recurrence.  They agree at $n=0,1,2$ by
		the initial values of $K$, $L$, $M$.  Uniqueness of recurrence
		solutions gives equality for all $n$.
	\end{proof}
	
	\begin{corollary}[Shift invariance]
		\label{cor:shift}
		For any weights $(k,l,m)$ and any integers $n$, $j$,
		\[
		f_{n+j}(k,l,m) \;=\; K(n)\,f_{j+2}(k,l,m) \;+\; L(n)\,f_{j+1}(k,l,m) \;+\; M(n)\,f_j(k,l,m).
		\]
	\end{corollary}
	\begin{proof}
		Fix $j$.  The shifted sequence $h_n = f_{n+j}(k,l,m)$ satisfies the
		same recurrence, with $h_0 = f_j$, $h_1 = f_{j+1}$, $h_2 = f_{j+2}$.
		Apply Theorem~\ref{thm:decomposition} to $h_n$.
	\end{proof}
	
	The Decomposition Theorem says: \emph{to evaluate any weighted power sum
		at index $n$, it suffices to know it at three consecutive indices — and
		the coefficients are always $K(n)$, $L(n)$, $M(n)$, regardless of the
		weights}.

	\section{Applications and Key Lemmas}
	\label{sec:applications}
	
	\subsection{Newton's identity: $s_n = x^n + y^n + z^n$}
	
	Apply the decomposition with weights $(1,1,1)$, noting
	$f_0(1,1,1)=3$, $f_1(1,1,1)=e_1$, $f_2(1,1,1)=e_1^2-2e_2$:
	\[
	s_n \;=\; K(n)\,(e_1^2-2e_2) \;+\; L(n)\,e_1 \;+\; 3\,M(n).
	\]
	In particular, $s_0=3$, $s_1=e_1$, $s_2=e_1^2-2e_2$ match the known
	Newton values, and equality for all $n$ follows.  Expanding the
	definitions yields the key formula:
	
	\begin{lemma}
		\label{lem:sn-formula}
		$s_n \;=\; e_1 K(n+1) - 2e_2 K(n) + 3e_3 K(n-1)$ for all integers $n$.
	\end{lemma}
	\begin{proof}
		Apply the Decomposition Theorem with weights $(1,1,1)$, substituting
		$f_0(1,1,1)=3$, $f_1(1,1,1)=e_1$, $f_2(1,1,1)=e_1^2-2e_2$ and the
		definitions $L(n) = K(n+1)-e_1 K(n)$, $M(n) = e_3 K(n-1)$:
		\[
		s_n = K(n)(e_1^2-2e_2) + L(n)\,e_1 + 3M(n)
		= K(n)(e_1^2-2e_2) + (K(n+1)-e_1 K(n))e_1 + 3e_3 K(n-1).
		\]
		Expanding and cancelling the $e_1^2 K(n)$ terms gives the result.
	\end{proof}
	
	\begin{theorem}
		\label{thm:sn-basis}
		$s_n \;=\; K(n+2) + L(n+1) + M(n)$ for all integers $n$.
	\end{theorem}
	\begin{proof}
		By Lemma~\ref{lem:sn-formula}, $s_n = e_1 K(n+1) - 2e_2 K(n) + 3e_3 K(n-1)$.
		Substituting $K(n+2) = e_1 K(n+1)-e_2 K(n)+e_3 K(n-1)$,
		$L(n+1) = K(n+2)-e_1 K(n+1)$, and $M(n) = e_3 K(n-1)$, one verifies
		that $K(n+2)+L(n+1)+M(n) = e_1 K(n+1) - 2e_2 K(n) + 3e_3 K(n-1) = s_n$.
	\end{proof}
	
	% -----------------------------------------------------------------------
	\section{Generalised $K$ and Substitution Identities}
	\label{sec:substitutions}
	
	\subsection{The generalised function $K_n(a,b,c)$}
	
	The formula in Definition~\ref{def:K} makes sense for any distinct
	nonzero triple $(a,b,c)$.
	
	\begin{definition}
		\label{def:Kgen}
		For distinct nonzero $a,b,c$ and any integer $n$,
		\[
		K_n(a,b,c) \;=\; \frac{\cyc{a^n(b-c)}}{\cyc{a^2(b-c)}}
		\;=\; \frac{a^n(b-c)+b^n(c-a)+c^n(a-b)}{-(b-c)(c-a)(a-b)}.
		\]
	\end{definition}
	
	So $K_n(x,y,z) = K(n)$.  The function $K_n(a,b,c)$ satisfies the
	recurrence with symmetric polynomials $\varepsilon_i$ of $(a,b,c)$ in
	place of $e_i$, and has initial values $K_0=K_1=0$, $K_2=1$ for any
	admissible triple.
	
	The power of this definition is that \emph{evaluating $K_n$ at a
		specific triple reduces, via the Decomposition Theorem, to a clean
		identity in $K$, $L$, $M$}.  The next three subsections illustrate
	this with the three most natural substitutions.
	
	\subsection{The product substitution: $(a,b,c) = (yz, zx, xy)$}
	
	The triple $(yz, zx, xy)$ has symmetric polynomials
	\[
	\varepsilon_1 = e_2, \qquad \varepsilon_2 = e_1 e_3, \qquad \varepsilon_3 = e_3^2.
	\]
	To evaluate $K_n(yz,zx,xy)$, note that by definition
	\[
	\Delta \cdot K_n(yz,zx,xy) \;=\; \cyc{(yz)^n(zx - xy)}
	\;=\; f_n\!\left(z^{n-1}-y^{n-1},\; x^{n-1}-z^{n-1},\; y^{n-1}-x^{n-1}\right).
	\]
	Applying the Decomposition Theorem with these weights:
	\begin{itemize}
		\item $f_0 = 0$ (the weights sum to zero),
		\item $f_1 = \cyc{x^{n-1}(y-z)} = K(n-1)\cdot\Delta$,
		\item $f_2 = \cyc{x^{n-1}(y^2-z^2)} = -L(n-1)\cdot\Delta$
		\quad (both sides satisfy the same recurrence and agree at $n=2,3,4$).
	\end{itemize}
	Therefore
	\[
	\Delta \cdot K_n(yz,zx,xy)
	= K(n)\cdot(-L(n-1)\Delta) + L(n)\cdot(K(n-1)\Delta),
	\]
	and dividing by $\Delta$:
	
	\begin{theorem}[Product-substitution identity]
		\label{thm:product}
		For all integers $n$,
		\[
		K_n(yz,\,zx,\,xy) \;=\; K(n)\,L(n-1) - L(n)\,K(n-1).
		\]
	\end{theorem}
	
	Substituting $L(k) = K(k+1) - e_1 K(k)$ and expanding, the $e_1$ terms cancel:
	
	\begin{corollary}[Cassini identity]
		\label{cor:cassini}
		\[
		K(n)^2 - K(n+1)\,K(n-1) \;=\; K_n(yz,\,zx,\,xy).
		\]
	\end{corollary}
	
	\begin{remark}
		This mirrors the classical Cassini identity $F_n^2 - F_{n+1}F_{n-1}=(-1)^{n-1}$
		for Fibonacci numbers, where the right-hand side is a constant (up to sign).
		Here the right-hand side is the richer object $K_n(yz,zx,xy)$,
		which satisfies its own third-order recurrence with coefficients
		$(e_2, e_1e_3, e_3^2)$.  In particular, $K(n)^2 - K(n+1)K(n-1)$
		is a polynomial in $e_2$, $e_1 e_3$, $e_3^2$ only — $e_1$ never appears alone.
	\end{remark}
	
	\subsection{The squaring substitution: $(a,b,c) = (x^2, y^2, z^2)$}
	
	The triple $(x^2,y^2,z^2)$ has symmetric polynomials
	\[
	\varepsilon_1 = e_1^2-2e_2, \qquad \varepsilon_2 = e_2^2-2e_1e_3, \qquad \varepsilon_3 = e_3^2.
	\]
	
	\begin{theorem}[Squaring-substitution identity]
		\label{thm:squaring}
		For all integers $n$,
		\[
		(e_3 - e_1 e_2)\cdot K_n(x^2,\,y^2,\,z^2) \;=\; L(2n).
		\]
	\end{theorem}
	\begin{proof}
		This is the special case $k=2$ of Theorem~\ref{thm:power} in
		Section~\ref{sec:power}, proved there in full.  We verify the
		prefactor: $\Delta_2/\Delta = (x+y)(y+z)(z+x) = -(e_3-e_1e_2)$,
		and at $k=2$ the right-hand side of Theorem~\ref{thm:power} is
		$K(2n)\,K(3) - K(2n+1)\,K(2) = e_1 K(2n) - K(2n+1) = -L(2n)$,
		since $K(2)=1$, $K(3)=e_1$, and $L(2n) = K(2n+1)-e_1 K(2n)$.
		Combining signs gives the stated identity.
	\end{proof}
	
	Note $e_3 - e_1 e_2 = -(x+y)(y+z)(z+x)$, the product of pairwise sums.

	\section{General Power Substitution}
	\label{sec:power}
	
	For an integer $k \geq 1$, write the \emph{$k$-th difference discriminant} as
	\[
	\Delta_k \;=\; -(y^k-z^k)(z^k-x^k)(x^k-y^k),
	\]
	consistently with the sign convention $\Delta = \Delta_1 = -(y-z)(z-x)(x-y)$.
	The ratio $\Delta_k/\Delta$ is a symmetric polynomial in $x,y,z$; for
	example $\Delta_2/\Delta = (x+y)(y+z)(z+x) = -(e_3 - e_1 e_2)$.
	
	\begin{theorem}[General power substitution]
		\label{thm:power}
		For all integers $k \geq 1$ and $n$,
		\[
		\frac{\Delta_k}{\Delta}\, K_n(x^k,y^k,z^k)
		\;=\;
		K(kn)\,K(k+1) - K(kn+1)\,K(k).
		\]
	\end{theorem}
	\begin{proof}
		By definition, the numerator of $K_n(x^k,y^k,z^k)$ is
		\[
		\cyc{(x^k)^n(y^k-z^k)} \;=\; f_{kn}(y^k-z^k,\, z^k-x^k,\, x^k-y^k),
		\]
		since $(x^k)^n \cdot (y^k - z^k)$ is the $x$-term of $f_{kn}$ with weights
		$(y^k-z^k, z^k-x^k, x^k-y^k)$.  Denote these weights $(p,q,r)$.
		We evaluate $f_0, f_1, f_2$ at $(p,q,r)$:
		\begin{itemize}
			\item $f_0(p,q,r) = 0$, since $p+q+r = 0$.
			\item $f_1(p,q,r) = \cyc{x(y^k-z^k)}$.  Expanding the cyclic sum
			and collecting the $x^k$ terms (each $x \cdot y^k$ pairs with
			$-y \cdot x^k$ from the next term):
			$\cyc{x(y^k-z^k)} = \cyc{x^k(z-y)} = -\cyc{x^k(y-z)} = -K(k)\,\Delta$.
			\item $f_2(p,q,r) = \cyc{x^2(y^k-z^k)} = -\cyc{x^k(y^2-z^2)}$.
			Writing $y^2-z^2 = (y-z)(y+z) = -w_x(y+z)$ and expanding,
			$\cyc{x^k(y^2-z^2)} = e_1\,f_k(w_x,w_y,w_z) - f_{k+1}(w_x,w_y,w_z)
			= (e_1 K(k) - K(k+1))\Delta = -L(k)\,\Delta$,
			so $f_2(p,q,r) = L(k)\,\Delta$.
		\end{itemize}
		Applying the Decomposition Theorem (with $kn$ in place of $n$):
		\[
		f_{kn}(p,q,r) \;=\; K(kn)\cdot L(k)\,\Delta \;+\; L(kn)\cdot(-K(k))\,\Delta,
		\]
		and since $f_{kn}(p,q,r) = K_n(x^k,y^k,z^k)\cdot\Delta_k$, dividing by $\Delta$:
		\[
		\frac{\Delta_k}{\Delta}\,K_n(x^k,y^k,z^k) = K(kn)\,L(k) - L(kn)\,K(k).
		\]
		Substituting $L(m) = K(m+1) - e_1 K(m)$ into both $L(k)$ and $L(kn)$ and
		expanding, the $e_1$ terms cancel, leaving $K(kn)\,K(k+1) - K(kn+1)\,K(k)$.
	\end{proof}
	
	\subsection*{Special case $n=1$}
	
	Since $K_1(a,b,c) = 0$ for any triple, the left-hand side vanishes, and
	the theorem gives $K(k)\,K(k+1) - K(k+1)\,K(k) = 0$: trivially true, and
	a consistency check.
	
	\subsection*{Special case $n=2$}
	
	Since $K_2(a,b,c) = 1$ for any triple, the theorem gives the closed formula
	\[
	\frac{\Delta_k}{\Delta}
	\;=\;
	K(2k)\,K(k+1) - K(2k+1)\,K(k).
	\]
	
	\subsection*{Special case $n=3$}
	
	For any triple $(a,b,c)$ one has $K_3(a,b,c) = a+b+c$ directly from the
	closed form: the numerator $\cyc{a^3(b-c)}$ factors as
	$(a+b+c)\cdot\cyc{a^2(b-c)}$.  Applied to $(x^k,y^k,z^k)$ this gives
	$K_3(x^k,y^k,z^k) = s_k$, and the theorem yields
	\[
	\frac{\Delta_k}{\Delta}\, s_k
	\;=\;
	K(3k)\,K(k+1) - K(3k+1)\,K(k).
	\]
	
	\subsection*{Recovery of the squaring identity (Theorem~\ref{thm:squaring})}
	
	Set $k=2$.  Then $K(2)=1$ and $L(2)=K(3)-e_1K(2)=e_1-e_1=0$, so
	$K(kn)\,L(k) - L(kn)\,K(k) = -L(2n)$.  Since
	$\Delta_2/\Delta = (x+y)(y+z)(z+x) = -(e_3-e_1e_2)$,
	Theorem~\ref{thm:power} gives
	\[
	-(e_3 - e_1 e_2)\cdot K_n(x^2,y^2,z^2) \;=\; -L(2n),
	\]
	which is exactly Theorem~\ref{thm:squaring}.

	\section{An Addition Formula for $K$}
	\label{sec:addition}
	
	The following identity is used in Section~\ref{sec:flt} to relate
	$K(3n)$ and $K(3n+1)$ to $K(n)$ and $K(n+1)$.
	
	\begin{lemma}[Addition formula]
		\label{lem:addition}
		For all integers $m$ and $n$,
		\[
		K(m+n) \;=\; K(m)\,K(n+1) - e_3\,K(m-1)\,K(n-1) + \bigl(K(m+1)-e_1 K(m)\bigr)\,K(n).
		\]
		Equivalently, using $L(m) = K(m+1)-e_1 K(m)$ and $M(m)=e_3 K(m-1)$,
		\[
		K(m+n) \;=\; K(m)\,K(n+1) + L(m)\,K(n) - M(m)\,K(n-1).
		\]
	\end{lemma}
	\begin{proof}
		Fix $m$ and consider both sides as functions of $n$.  Both satisfy the
		standard recurrence (in $n$) with coefficients $e_1, e_2, e_3$.  At
		$n = 0, 1, 2$ the right-hand side gives, respectively,
		\[
		K(m+0)=K(m),\quad K(m+1),\quad K(m+2),
		\]
		matching the left-hand side.  Uniqueness of recurrence solutions
		(Theorem~\ref{thm:decomposition}) gives equality for all $n$.
	\end{proof}
	
	
	% -----------------------------------------------------------------------
	\section{Consequences of $s_n = 0$ over the reals}
	\label{sec:sn-zero}
	
	Throughout this section $x$, $y$, $z$ are distinct nonzero real (or
	complex) numbers satisfying $s_n = x^n + y^n + z^n = 0$ for a fixed
	integer $n \geq 3$.  No integrality is assumed.
	
	\subsection*{The ratio identity}
	
	Setting $k = n$ and $m = 3$ in Theorem~\ref{thm:power} and using
	$K_3(x^n, y^n, z^n) = s_n = 0$ gives immediately:
	
	\begin{corollary}
		\label{cor:sn-ratio}
		If $s_n = 0$ then $K(3n)\,K(n+1) = K(3n+1)\,K(n)$.
	\end{corollary}
	
	In other words the two ratios $K(3n)/K(n)$ and $K(3n+1)/K(n+1)$ are
	equal (whenever the denominators are nonzero).  We write their common
	value $A = K(3n)/K(n)$.
	
	\subsection*{The common ratio $A$}
	
	\begin{theorem}
		\label{thm:sn-ratio}
		If $s_n = 0$ and $K(n) \neq 0$, define $A = K(3n)/K(n)$.  Then
		\[
		K(3n+1) = A\,K(n+1).
		\]
	\end{theorem}
	\begin{proof}
		The ratio identity (Corollary~\ref{cor:sn-ratio}) gives
		$K(3n)\,K(n+1) = K(3n+1)\,K(n)$.  Substituting $K(3n) = A\,K(n)$
		and dividing both sides by $K(n)$ yields the result.
	\end{proof}
	
	\subsection*{The master identity}
	
	The two equalities above are the $j = 0$ and $j = 1$ cases of a
	uniform identity covering every offset $j$.
	
	\begin{theorem}[Master identity]
		\label{thm:master}
		If $s_n = 0$ and $K(n) \neq 0$, then for all integers $j$,
		\[
		K(3n+j) \;=\; A\,K(n+j) \;+\; e_3^n\,K(j),
		\]
		where $A = K(3n)/K(n)$.
	\end{theorem}
	\begin{proof}
		Fix $n$.  The right-hand side, viewed as a function of $j$, is a sum
		of two sequences: $j \mapsto A\,K(n+j)$ (a scalar multiple of a shift
		of $K$) and $j \mapsto e_3^n\,K(j)$ (a scalar multiple of $K$ itself).
		Both satisfy the recurrence with coefficients $e_1, e_2, e_3$, and
		therefore so does their sum.  Since the left-hand side also satisfies
		the same recurrence, it suffices to check agreement at $j = 0, 1, 2$.
		
		\smallskip\noindent
		$j = 0$: $A\,K(n) + e_3^n\cdot 0 = A\,K(n) = K(3n)$. \checkmark
		
		\smallskip\noindent
		$j = 1$: $A\,K(n+1) + e_3^n\cdot 0 = A\,K(n+1) = K(3n+1)$,
		by Theorem~\ref{thm:sn-ratio}. \checkmark
		
		\smallskip\noindent
		$j = 2$: we need $K(3n+2) = A\,K(n+2) + e_3^n$.
		Apply the $s_n$ formula (Lemma~\ref{lem:sn-formula}) at index $3n$:
		\[
		s_{3n} = e_1\,K(3n+1) - 2e_2\,K(3n) + 3e_3\,K(3n-1).
		\]
		Substituting $K(3n) = A\,K(n)$, $K(3n+1) = A\,K(n+1)$, and using
		$s_n = 0$ to replace $e_1\,K(n+1) - 2e_2\,K(n) = -3e_3\,K(n-1)$:
		\[
		s_{3n} = 3e_3\,\bigl(K(3n-1) - A\,K(n-1)\bigr).
		\]
		Since $x^n + y^n + z^n = 0$, Newton's identity applied to
		$(x^n, y^n, z^n)$ gives $s_{3n} = 3(xyz)^n = 3e_3^n$, so
		$K(3n-1) = A\,K(n-1) + e_3^{n-1}$.  Now the recurrence at $3n+2$:
		\begin{align*}
			K(3n+2)
			&= e_1\cdot A\,K(n+1) - e_2\cdot A\,K(n)
			+ e_3\,(A\,K(n-1) + e_3^{n-1}) \\
			&= A\,\bigl(e_1\,K(n+1) - e_2\,K(n) + e_3\,K(n-1)\bigr) + e_3^n
			= A\,K(n+2) + e_3^n. \quad\checkmark
		\end{align*}
		Uniqueness of recurrence solutions gives equality for all $j$.
	\end{proof}
	
	\begin{remark}
		The values $K(0) = K(1) = 0$ are what make $j = 0$ and $j = 1$
		trivially correct: the correction term $e_3^n\,K(j)$ simply vanishes
		there, so no computation is needed.  This has nothing to do with
		$s_n = 0$; it is a permanent feature of $K$ inherited from the
		antisymmetric weights $(y-z, z-x, x-y)$.  The assumption $s_n = 0$
		contributes exactly one thing: it makes the ratio $K(3n)/K(n)$ equal
		to $K(3n+1)/K(n+1)$, promoting a per-index quantity into the single
		global constant $A$ that the identity requires.  Without $s_n = 0$
		no such $A$ exists.
	\end{remark}
	
	\begin{remark}
		The master identity can be read as a decomposition of the $3n$-shift:
		\begin{itemize}
			\item $A \cdot K(n+j)$: a scaled copy of the original sequence,
			\item $e_3^n \cdot K(j)$: a copy of $K$ scaled by $(xyz)^n$,
			encoding $s_{3n} = 3e_3^n \neq 0$.
		\end{itemize}
		The natural basis window for this shift is $\{K(3n), K(3n+1), K(3n+2)\}$
		(indices $j = 0, 1, 2$), since it is the standard initial window of $K$
		translated by $3n$.  Within it, the correction $(0, 0, e_3^n)$ mirrors
		exactly the initial condition $(K(0), K(1), K(2)) = (0, 0, 1)$.
	\end{remark}

	\section{Constraints on a hypothetical FLT counter-example}
	\label{sec:flt}
	
	\noindent\textit{Disclaimer.}
	Fermat's Last Theorem was proved by Wiles in 1995, and nothing here
	comes close to an independent proof --- nor is that the aim.  The goal
	is to see what arithmetic constraints the framework imposes on a
	hypothetical primitive solution $x^n + y^n + z^n = 0$ with
	$x, y, z \in \mathbb{Z}$.
	
	\medskip
	Throughout this section $x$, $y$, $z$ are pairwise coprime nonzero
	integers with $s_n = 0$ and $n \geq 3$ an odd prime.  All results of
	Section~\ref{sec:sn-zero} apply; in particular the ratio $A$ and the
	master identity are in force.  The additional content here is purely
	about divisibility.
	
	\subsection*{Divisibility consequences of the master identity}
	
	Since $x$, $y$, $z$ are integers, every $K(m)$ is an integer.
	We now show $K(n) \mid K(3n)$, after which divisibility of $K(3n+1)$
	and the full master identity follow with no further calculation.
	
	\begin{theorem}
		\label{thm:flt-divisibility}
		If $x$, $y$, $z$ are coprime integers with $s_n = 0$, then
		$K(n) \mid K(3n)$, and consequently $K(n+1) \mid K(3n+1)$, with
		the same integer quotient $A = K(3n)/K(n)$.
	\end{theorem}
	\begin{proof}
		We show $K(3n) \equiv 0 \pmod{K(n)}$.  Applying the addition formula
		(Lemma~\ref{lem:addition}) with $m = n$ and working modulo $K(n)$:
		\begin{align*}
			K(2n)   &\equiv L(n)\,K(n+1), \\
			K(2n+1) &\equiv L(n)\,K(n+2) + M(n)\,K(n+1),
		\end{align*}
		since every term containing $K(n)$ as a factor vanishes.  A further
		application modulo $K(n)$, substituting
		$L(n) \equiv K(n+1)$, $M(n) \equiv e_3\,K(n-1)$,
		and $K(n+2) \equiv e_1\,K(n+1) + e_3\,K(n-1)$, gives
		\[
		K(3n) \;\equiv\; K(n+1)^2\,\bigl(e_1\,K(n+1) + 3e_3\,K(n-1)\bigr) \pmod{K(n)}.
		\]
		Since $s_n = 0$, Lemma~\ref{lem:sn-formula} gives
		$e_1\,K(n+1) + 3e_3\,K(n-1) = 2e_2\,K(n) \equiv 0 \pmod{K(n)}$.
		Hence $K(3n) \equiv 0 \pmod{K(n)}$, so $A = K(3n)/K(n)$ is an integer.
		
		Since $A$ is an integer, Theorem~\ref{thm:sn-ratio} gives
		$K(3n+1) = A\,K(n+1)$, so $K(n+1) \mid K(3n+1)$ with the same
		quotient $A$.
	\end{proof}
	
	The master identity (Theorem~\ref{thm:master}) then gives integer
	divisibility at every offset: $K(n) \mid K(3n+j) - e_3^n\,K(j)$
	for all integers $j$.
	
	\subsection*{Bounding $d = \gcd(K(n),\, K(n+1))$}
	
	We use the identities
	\begin{align}
		s_n     &= e_1\,K(n+1) - 2e_2\,K(n) + 3e_3\,K(n-1), \label{eq:sn}\\
		3K(n+2) &= 3e_1\,K(n+1) - 3e_2\,K(n) + 3e_3\,K(n-1). \label{eq:3Kn2}
	\end{align}
	Since $s_n = 0$ and $d \mid K(n), K(n+1)$, equation~\eqref{eq:sn} gives
	\begin{equation}
		\label{eq:d-3e3}
		d \;\mid\; 3e_3\,K(n-1).
	\end{equation}
	Substituting into~\eqref{eq:3Kn2} gives $d \mid 3K(n+2)$.  So $d$
	divides $3K(n)$, $3K(n+1)$, $3K(n+2)$: three consecutive multiples of
	$3$.  Applying the recurrence inductively, $d \mid 3K(m)$ for all
	$m \geq n$, and in particular $d \mid 3K(2n-1)$, $3K(2n)$, $3K(2n+1)$.
	
	Applying~\eqref{eq:sn} at index $2n$ gives $d \mid s_{2n}$.  Since
	$s_n = 0$,
	\[
	s_{2n} = s_n^2 - 2\cyc{(xy)^n} = -2(x^ny^n + y^nz^n + z^nx^n),
	\]
	so
	\begin{equation}
		\label{eq:d-cross}
		d \;\mid\; 2(x^ny^n + y^nz^n + z^nx^n).
	\end{equation}
	
	\medskip\noindent\textbf{Parity.}
	Since $x$, $y$, $z$ are pairwise coprime with $s_n = 0$, exactly one is
	even: if all three were odd then $s_n \equiv 1+1+1 = 3 \equiv 1 \pmod{2}
	\neq 0$, and if two were even they would share the factor~$2$,
	contradicting coprimeness.  Therefore $e_1, e_3$ are even and $e_2$ is
	odd.  The recurrence mod~$2$ becomes $K(m+3) \equiv K(m+1) \pmod{2}$,
	giving the alternating pattern $K(\text{even}) \equiv 1$,
	$K(\text{odd}) \equiv 0 \pmod{2}$.  Since $n$ is an odd prime, $K(n)$
	is even and $K(n+1)$ is odd, so $2 \nmid d$.
	
	\medskip
	We now analyse which odd prime powers can divide $d$.  Let $p$ be any
	odd prime dividing $d$, and let $a = p^k$ be the exact power of $p$ in
	$d$.
	
	\medskip\noindent\textbf{Case 1: $p \mid e_3 = xyz$.}
	Since $x$, $y$, $z$ are pairwise coprime, $p$ divides exactly one of
	them, say $p \mid x$, so $a \mid x^n$.  From~\eqref{eq:d-cross},
	$a \mid 2(x^ny^n + y^nz^n + z^nx^n)$.  Since $a \mid x^n$ the first and
	third cross terms vanish mod $a$, leaving $a \mid 2y^nz^n$.  Since $p$
	is odd, $\gcd(a,2)=1$, and $\gcd(x,y)=\gcd(x,z)=1$ gives
	$\gcd(a,y^nz^n)=1$, so $a \mid 1$: contradiction.  No odd prime factor
	of $d$ divides $e_3$.
	
	\medskip\noindent\textbf{Case 2: $p \nmid e_3$.}
	From~\eqref{eq:d-3e3}, $a \mid 3e_3 K(n-1)$.  Since $p \nmid e_3$,
	we get $a \mid 3K(n-1)$.  Together with $a \mid 3K(n)$ and
	$a \mid 3K(n+1)$, backward induction on the recurrence
	\[
	3K(m) = 3e_1\,K(m+1) - 3e_2\,K(m+2) + 3e_3\,K(m+3)
	\]
	(valid at each step since $p \nmid e_3$) gives $a \mid 3K(m)$ for all
	$m \leq n$.  At $m = 2$: $a \mid 3K(2) = 3$, so $p^k \mid 3$, hence
	$p = 3$ and $k = 1$.
	
	\medskip
	Combining: $d$ is odd (parity), Case~1 is vacuous for odd primes, and
	Case~2 forces every prime power in $d$ to divide $3$.
	
	\begin{proposition}
		\label{prop:d-divides-3}
		If $x$, $y$, $z$ are pairwise coprime nonzero integers with $s_n = 0$
		and $n$ an odd prime, then $\gcd(K(n),\, K(n+1))$ divides $3$.
	\end{proposition}
	
	\begin{remark}
		Working with the exact prime power $a = p^k$ handles higher powers
		such as $9$ in the same stroke: $p^k \mid 3$ forces $k = 1$
		immediately, with no separate argument needed.
	\end{remark}
