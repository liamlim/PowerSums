	\section{Setup}
	
	Throughout, $x$, $y$, $z$ are distinct and nonzero complex (or real)
	numbers.  We write their elementary symmetric polynomials as
	\[
	e_1 = x+y+z, \qquad e_2 = xy+yz+zx, \qquad e_3 = xyz,
	\]
	and define the \emph{minimal polynomial}
	\[
	P(t) = t^3 - e_1 t^2 + e_2 t - e_3 = (t-x)(t-y)(t-z).
	\]
	Since $e_3 = xyz \neq 0$ we may divide by any power of $x$, $y$, or $z$,
	and in particular extend all recurrences to all integers $n \in \mathbb{Z}$.
	
	\subsection*{Cyclic sum notation}
	
	Many expressions in this paper are cyclic in $x, y, z$.  We write
	\[
	\cyc{g(x)} \;:=\; g(x) + g(y) + g(z)
	\]
	for the cyclic sum of any expression $g$, where the implicit
	substitution rule is $x \to y \to z \to x$.  For example,
	$e_1 = \cyc{x}$, $e_2 = \cyc{xy}$, $e_3 = xyz$, and more usefully:
	\[
	\cyc{x^n(y-z)} \;=\; x^n(y-z) + y^n(z-x) + z^n(x-y).
	\]
	This notation eliminates the need to write all three terms of any
	cyclic identity.

	\section{Weighted Power Sums}
	
	\begin{definition}
		For any integer $n$ and a weight triple $(k, l, m)$, define
		\[
		f_n(k,l,m) = k x^n + l y^n + m z^n = \cyc{k\, x^n},
		\]
		where in the cyclic notation it is understood that the weights cycle
		as $(k,l,m) \to (l,m,k)$ alongside $(x,y,z) \to (y,z,x)$.
	\end{definition}
	
	Special cases of immediate interest:
	\[
	s_n \;:=\; f_n(1,1,1) \;=\; x^n + y^n + z^n \;=\; \cyc{x^n}.
	\]
	
	The key properties of $f_n$ are linearity in weights and the universal
	recurrence inherited from $P(t)$.
	
	\begin{theorem}[Recurrence]
		\label{thm:recurrence}
		For any fixed weights $(k,l,m)$ and any integer $n$,
		\[
		f_{n+3}(k,l,m) = e_1\, f_{n+2}(k,l,m) - e_2\, f_{n+1}(k,l,m) + e_3\, f_n(k,l,m).
		\]
	\end{theorem}
	\begin{proof}
		Each of $x,y,z$ satisfies $t^3 = e_1 t^2 - e_2 t + e_3$.  Multiply by
		$t^n$, weight by $k, l, m$ respectively, and sum.
	\end{proof}
	
	Because $e_3 \neq 0$ the recurrence can also be run backwards, making
	$f_n(k,l,m)$ well-defined for all $n \in \mathbb{Z}$.

	\section{The Basis $K$, $L$, $M$}
	
	Any sequence satisfying the recurrence of Theorem~\ref{thm:recurrence}
	is determined by its values at three consecutive integers.  We single
	out the three basis sequences with initial values forming the identity
	matrix at $n = 0, 1, 2$.
	
	\subsection{Definition via the antisymmetric weights}
	
	The triple of \emph{difference weights}
	\[
	(w_x,\, w_y,\, w_z) \;:=\; (y-z,\; z-x,\; x-y)
	\]
	satisfies $w_x + w_y + w_z = 0$ and is nonzero since $x,y,z$ are
	distinct.  A short computation gives
	\[
	f_0(w_x,w_y,w_z) = 0, \quad f_1(w_x,w_y,w_z) = 0, \quad
	f_2(w_x,w_y,w_z) = \cyc{x^2(y-z)} =: \Delta,
	\]
	where $\Delta = -(y-z)(z-x)(x-y) \neq 0$.  (The $f_2$ identity follows
	by straightforward algebra, or see Section~\ref{sec:substitutions}.)
	
	\begin{definition}
		\label{def:K}
		For any integer $n$, define
		\[
		K(n) \;=\; \frac{f_n(w_x,\,w_y,\,w_z)}{\Delta}
		\;=\; \frac{\cyc{x^n(y-z)}}{\cyc{x^2(y-z)}}.
		\]
	\end{definition}
	
	\begin{definition}
		\label{def:LM}
		\[
		L(n) = K(n+1) - e_1\,K(n), \qquad M(n) = e_3\,K(n-1).
		\]
	\end{definition}
	
	The initial values follow immediately:
	
	\[
	\renewcommand{\arraystretch}{1.3}
	\begin{array}{c|ccc}
		n & 0 & 1 & 2 \\\hline
		K(n) & 0 & 0 & 1 \\
		L(n) & 0 & 1 & 0 \\
		M(n) & 1 & 0 & 0
	\end{array}
	\]
	
	The triple $(K, L, M)$ forms the \emph{standard basis} for all integer
	sequences satisfying the recurrence: the columns at $n=0,1,2$ are
	exactly the identity matrix.  All three sequences satisfy the same
	recurrence as $f_n$, since $K$ inherits it from $f_n$, and $L$, $M$
	are linear combinations of shifts of $K$.
	
	\begin{remark}
		The explicit values of $K$ at small $n$ are
		$K(2)=1$, $K(3)=e_1$, $K(4)=e_1^2-e_2$, and so on; $L$ and $M$ are
		recovered from $K$ via the definitions above.  The backward extension
		gives $K(-1) = 1/e_3$, consistent with $M(0) = e_3 K(-1) = 1$.
	\end{remark}

	\section{The Decomposition Theorem}
	\label{sec:decomposition}
	
	The following is the central engine of everything that follows.
	
	\begin{theorem}[Decomposition]
		\label{thm:decomposition}
		For any weights $(k,l,m)$ and any integer $n$,
		\[
		f_n(k,l,m) \;=\; K(n)\,f_2(k,l,m) \;+\; L(n)\,f_1(k,l,m) \;+\; M(n)\,f_0(k,l,m).
		\]
	\end{theorem}
	\begin{proof}
		Both sides satisfy the same recurrence.  They agree at $n=0,1,2$ by
		the initial values of $K$, $L$, $M$.  Uniqueness of recurrence
		solutions gives equality for all $n$.
	\end{proof}
	
	\begin{corollary}[Shift invariance]
		\label{cor:shift}
		For any weights $(k,l,m)$ and any integers $n$, $j$,
		\[
		f_{n+j}(k,l,m) \;=\; K(n)\,f_{j+2}(k,l,m) \;+\; L(n)\,f_{j+1}(k,l,m) \;+\; M(n)\,f_j(k,l,m).
		\]
	\end{corollary}
	\begin{proof}
		Fix $j$.  The shifted sequence $h_n = f_{n+j}(k,l,m)$ satisfies the
		same recurrence, with $h_0 = f_j$, $h_1 = f_{j+1}$, $h_2 = f_{j+2}$.
		Apply Theorem~\ref{thm:decomposition} to $h_n$.
	\end{proof}
	
	The Decomposition Theorem says: \emph{to evaluate any weighted power sum
		at index $n$, it suffices to know it at three consecutive indices — and
		the coefficients are always $K(n)$, $L(n)$, $M(n)$, regardless of the
		weights}.

	\section{First Applications}
	\label{sec:applications}
	
	\subsection{Newton's identity: $s_n = x^n + y^n + z^n$}
	
	Apply the decomposition with weights $(1,1,1)$, noting
	$f_0(1,1,1)=3$, $f_1(1,1,1)=e_1$, $f_2(1,1,1)=e_1^2-2e_2$:
	\[
	s_n \;=\; K(n)\,(e_1^2-2e_2) \;+\; L(n)\,e_1 \;+\; 3\,M(n).
	\]
	In particular, $s_0=3$, $s_1=e_1$, $s_2=e_1^2-2e_2$ match the known
	Newton values, and equality for all $n$ follows.  This gives the
	elegant identity:
	
	\begin{theorem}
		\label{thm:sn}
		$s_n \;=\; K(n+2) + L(n+1) + M(n)$ for all integers $n$.
	\end{theorem}
	\begin{proof}
		The right-hand side equals $T_n(1,1,1)$ where
		$T_n(k,l,m) = k\,K(n+2)+l\,L(n+1)+m\,M(n)$.  One checks that
		$T_n(1,1,1)$ and $s_n$ agree at $n=0,1,2$ and satisfy the same
		recurrence.
	\end{proof}
	
	\begin{theorem}
		$s_n \;=\; e_1 K(n+1) - 2e_2 K(n) + 3e_3 K(n-1)$ for all integers $n$.
	\end{theorem}
	
	\subsection{A relation between $K$, $L$, $M$}
	
	Matching $T_n(k,l,m) = K(n)$ at $n=0,1,2$ uniquely determines
	$(k,l,m) = (0, -1/e_2, 1/e_2)$, yielding:
	
	\begin{corollary}
		\label{cor:KLM}
		$e_2\,K(n) = M(n) - L(n+1)$ for all integers $n$.
	\end{corollary}

	\section{Generalised $K$ and Substitution Identities}
	\label{sec:substitutions}
	
	\subsection{The generalised function $K_n(a,b,c)$}
	
	The formula in Definition~\ref{def:K} makes sense for any distinct
	nonzero triple $(a,b,c)$.
	
	\begin{definition}
		\label{def:Kgen}
		For distinct nonzero $a,b,c$ and any integer $n$,
		\[
		K_n(a,b,c) \;=\; \frac{\cyc{a^n(b-c)}}{\cyc{a^2(b-c)}}
		\;=\; \frac{a^n(b-c)+b^n(c-a)+c^n(a-b)}{-(b-c)(c-a)(a-b)}.
		\]
	\end{definition}
	
	So $K_n(x,y,z) = K(n)$.  The function $K_n(a,b,c)$ satisfies the
	recurrence with symmetric polynomials $\varepsilon_i$ of $(a,b,c)$ in
	place of $e_i$, and has initial values $K_0=K_1=0$, $K_2=1$ for any
	admissible triple.
	
	The power of this definition is that \emph{evaluating $K_n$ at a
		specific triple reduces, via the Decomposition Theorem, to a clean
		identity in $K$, $L$, $M$}.  The next three subsections illustrate
	this with the three most natural substitutions.
	
	\subsection{The product substitution: $(a,b,c) = (yz, zx, xy)$}
	
	The triple $(yz, zx, xy)$ has symmetric polynomials
	\[
	\varepsilon_1 = e_2, \qquad \varepsilon_2 = e_1 e_3, \qquad \varepsilon_3 = e_3^2.
	\]
	To evaluate $K_n(yz,zx,xy)$, note that by definition
	\[
	\Delta \cdot K_n(yz,zx,xy) \;=\; \cyc{(yz)^n(zx - xy)}
	\;=\; f_n\!\left(z^{n-1}-y^{n-1},\; x^{n-1}-z^{n-1},\; y^{n-1}-x^{n-1}\right).
	\]
	Applying the Decomposition Theorem with these weights:
	\begin{itemize}
		\item $f_0 = 0$ (the weights sum to zero),
		\item $f_1 = \cyc{x^{n-1}(y-z)} = K(n-1)\cdot\Delta$,
		\item $f_2 = \cyc{x^{n-1}(y^2-z^2)} = -L(n-1)\cdot\Delta$
		\quad (both sides satisfy the same recurrence and agree at $n=2,3,4$).
	\end{itemize}
	Therefore
	\[
	\Delta \cdot K_n(yz,zx,xy)
	= K(n)\cdot(-L(n-1)\Delta) + L(n)\cdot(K(n-1)\Delta),
	\]
	and dividing by $\Delta$:
	
	\begin{theorem}[Product-substitution identity]
		\label{thm:product}
		For all integers $n$,
		\[
		K_n(yz,\,zx,\,xy) \;=\; K(n)\,L(n-1) - L(n)\,K(n-1).
		\]
	\end{theorem}
	
	Substituting $L(k) = K(k+1) - e_1 K(k)$ and expanding, the $e_1$ terms cancel:
	
	\begin{corollary}[Cassini identity]
		\label{cor:cassini}
		\[
		K(n)^2 - K(n+1)\,K(n-1) \;=\; K_n(yz,\,zx,\,xy).
		\]
	\end{corollary}
	
	\begin{remark}
		This mirrors the classical Cassini identity $F_n^2 - F_{n+1}F_{n-1}=(-1)^{n-1}$
		for Fibonacci numbers, where the right-hand side is a constant (up to sign).
		Here the right-hand side is the richer object $K_n(yz,zx,xy)$,
		which satisfies its own third-order recurrence with coefficients
		$(e_2, e_1e_3, e_3^2)$.  In particular, $K(n)^2 - K(n+1)K(n-1)$
		is a polynomial in $e_2$, $e_1 e_3$, $e_3^2$ only — $e_1$ never appears alone.
	\end{remark}
	
	\subsection{The squaring substitution: $(a,b,c) = (x^2, y^2, z^2)$}
	
	The triple $(x^2,y^2,z^2)$ has symmetric polynomials
	\[
	\varepsilon_1 = e_1^2-2e_2, \qquad \varepsilon_2 = e_2^2-2e_1e_3, \qquad \varepsilon_3 = e_3^2.
	\]
	A similar decomposition argument (both sides vanish at $n=0,1$, agree
	at $n=2$, and satisfy compatible recurrences) yields:
	
	\begin{theorem}[Squaring-substitution identity]
		\label{thm:squaring}
		For all integers $n$,
		\[
		(e_3 - e_1 e_2)\cdot K_n(x^2,\,y^2,\,z^2) \;=\; L(2n).
		\]
	\end{theorem}
	
	Note $e_3 - e_1 e_2 = -(x+y)(y+z)(z+x)$, the product of pairwise sums.

	\section{General Power Substitution}
	
	For an integer $k \geq 1$, write the \emph{$k$-th difference discriminant} as
	\[
	\Delta_k \;=\; -(y^k-z^k)(z^k-x^k)(x^k-y^k),
	\]
	consistently with the sign convention $\Delta = \Delta_1 = -(y-z)(z-x)(x-y)$.
	The ratio $\Delta_k/\Delta$ is a symmetric polynomial in $x,y,z$; for
	example $\Delta_2/\Delta = (x+y)(y+z)(z+x) = -(e_3 - e_1 e_2)$.
	
	\begin{theorem}[General power substitution]
		\label{thm:power}
		For all integers $k \geq 1$ and $n$,
		\[
		\frac{\Delta_k}{\Delta}\, K_n(x^k,y^k,z^k)
		\;=\;
		K(kn)\,K(k+1) - K(kn+1)\,K(k).
		\]
	\end{theorem}
	\begin{proof}
		By definition, the numerator of $K_n(x^k,y^k,z^k)$ is
		\[
		\cyc{(x^k)^n(y^k-z^k)} \;=\; f_{kn}(y^k-z^k,\, z^k-x^k,\, x^k-y^k),
		\]
		since $(x^k)^n \cdot (y^k - z^k)$ is the $x$-term of $f_{kn}$ with weights
		$(y^k-z^k, z^k-x^k, x^k-y^k)$.  Denote these weights $(p,q,r)$.
		We evaluate $f_0, f_1, f_2$ at $(p,q,r)$:
		\begin{itemize}
			\item $f_0(p,q,r) = 0$, since $p+q+r = 0$.
			\item $f_1(p,q,r) = \cyc{x(y^k-z^k)}$. Regrouping by $x^k$:
			$\cyc{x^k(z-y)} = -\cyc{x^k(y-z)} = -K(k)\,\Delta$.
			\item $f_2(p,q,r) = \cyc{x^2(y^k-z^k)} = -\cyc{x^k(y^2-z^2)}$.
			Writing $y^2-z^2 = (y-z)(y+z) = -w_x(y+z)$ and expanding,
			$\cyc{x^k(y^2-z^2)} = e_1\,f_k(w_x,w_y,w_z) - f_{k+1}(w_x,w_y,w_z)
			= (e_1 K(k) - K(k+1))\Delta = -L(k)\,\Delta$,
			so $f_2(p,q,r) = L(k)\,\Delta$.
		\end{itemize}
		Applying the Decomposition Theorem (with $kn$ in place of $n$):
		\[
		f_{kn}(p,q,r) \;=\; K(kn)\cdot L(k)\,\Delta \;+\; L(kn)\cdot(-K(k))\,\Delta,
		\]
		and since $f_{kn}(p,q,r) = K_n(x^k,y^k,z^k)\cdot\Delta_k$, dividing by $\Delta$:
		\[
		\frac{\Delta_k}{\Delta}\,K_n(x^k,y^k,z^k) = K(kn)\,L(k) - L(kn)\,K(k).
		\]
		Substituting $L(m) = K(m+1) - e_1 K(m)$ into both $L(k)$ and $L(kn)$ and
		expanding, the $e_1$ terms cancel, leaving $K(kn)\,K(k+1) - K(kn+1)\,K(k)$.
	\end{proof}
	
	\subsection*{Special case $n=1$}
	
	Since $K_1(a,b,c) = 0$ for any triple, the left-hand side vanishes, and
	the theorem gives $K(k)\,K(k+1) - K(k+1)\,K(k) = 0$: trivially true, and
	a consistency check.
	
	\subsection*{Special case $n=2$}
	
	Since $K_2(a,b,c) = 1$ for any triple, the theorem gives the closed formula
	\[
	\frac{\Delta_k}{\Delta}
	\;=\;
	K(2k)\,K(k+1) - K(2k+1)\,K(k).
	\]
	
	\subsection*{Special case $n=3$}
	
	For any triple $(a,b,c)$ one has $K_3(a,b,c) = a+b+c$ directly from the
	closed form: the numerator $\cyc{a^3(b-c)}$ factors as
	$(a+b+c)\cdot\cyc{a^2(b-c)}$.  Applied to $(x^k,y^k,z^k)$ this gives
	$K_3(x^k,y^k,z^k) = s_k$, and the theorem yields
	\[
	\frac{\Delta_k}{\Delta}\, s_k
	\;=\;
	K(3k)\,K(k+1) - K(3k+1)\,K(k).
	\]
	
	\subsection*{Recovery of the squaring identity (Theorem~\ref{thm:squaring})}
	
	Set $k=2$.  Then $K(2)=1$ and $L(2)=K(3)-e_1K(2)=e_1-e_1=0$, so
	$K(kn)\,L(k) - L(kn)\,K(k) = -L(2n)$.  Since
	$\Delta_2/\Delta = (x+y)(y+z)(z+x) = -(e_3-e_1e_2)$,
	Theorem~\ref{thm:power} gives
	\[
	-(e_3 - e_1 e_2)\cdot K_n(x^2,y^2,z^2) \;=\; -L(2n),
	\]
	which is exactly Theorem~\ref{thm:squaring}.

	\section{Exploring FLT counter-examples}
	In this section let's assume $x$,$y$,$z$ are distinct and non-zero integers such that
	$$s_n = 0$$
	
	According to the general power substitution theorem, we have
	\[
	\frac{\Delta_n}{\Delta}\, s_n
	\;=\;
	K(3n)\,K(n+1) - K(3n+1)\,K(n).
	\]
	
	So our choice of $x$,$y$,$z$ satisfies the simple formula 
	$$K(3n)\cdot K(n+1) = K(3n+1)\cdot K(n)$$
	
	Let's now investigate what are requirements on $d = \gcd(K(n), K(n+1))$
	
	We know that
	$$s_n = 0 = e_1 K(n+1) - 2e_2 K(n) + 3e_3 K(n-1)$$
	
	Also we know that $d$ doesn't divide $K(n-1)$ because otherwise there would be a common divisor of three consecutive $K$, which would result in $d$ dividing $K(k)$ for every $k$ including $K(2) = 1$ which is a contradiction.
	
	As such we can conclude that $d$ divides $3e_3$
	
	Also we know that
	$$K(2n+1) = K(n) K(n+3) + L(n) K(n+2) + M(n) K(n+1)$$
	$$K(2n) = K(n) K(n+2) + L(n) K(n+1) + M(n) K(n)$$
	
	It's trivial that $d$ divides $L(n) = K(n+1) - e_1 K(n)$ thus $d$ divides both $K(2n)$ and $K(2n+1)$
	
	Also we know that
	$$s_{2n} = 0 = e_1 K(2n+1) - 2e_2 K(2n) + 3e_3 K(2n-1)$$
	
	Together with the fact that $d$ divides $3e_3$ which we found earlier, we know that $d$ divides $s_{2n}$.
	
	Using $s_n = 0$ we know that
	
	$$s_{2n} = -2(x^ny^n + y^nz^n + z^nx^n)\equiv 0 \mod d$$
	$$3e_3 = 3xyz\equiv 0 \mod d$$
	
	It immediately follows that $x^ny^n + y^nz^n + z^nx^n$ and $xyz$ are coprime. As such we know that $d$ divides $6$.
	
	Now let's look back at formula 
	$$K(3n) K(n+1) = K(3n+1) K(n)$$
	
	Now that we know $\gcd(K(n), K(n+1))$ divides 6, we can conclude:
	$$6K(3n) \equiv 0 \mod K(n)$$
	$$6K(3n+1) \equiv 0 \mod K(n+1)$$
