	% -----------------------------------------------------------------------
	\section*{Introduction}
	
	This note develops a self-contained algebraic framework for power sums
	$s_n = x^n + y^n + z^n$ and their relatives.  The central object is a
	sequence $K(n)$ built from the alternating difference weights $(y-z, z-x, x-y)$; together
	with two companion sequences $L$ and $M$, it forms a basis for all sequences
	satisfying the same linear recurrence as the power sums.
	
	All arguments use only the recurrence and elementary algebra; no
	advanced number theory or analysis is required.
	
	% -----------------------------------------------------------------------
	\section{Setup}
	
	Throughout, $x$, $y$, $z$ are distinct and nonzero complex (or real)
	numbers.  We write their elementary symmetric polynomials as
	\[
	e_1 = x+y+z, \qquad e_2 = xy+yz+zx, \qquad e_3 = xyz,
	\]
	and define the \emph{minimal polynomial}
	\[
	P(t) = t^3 - e_1 t^2 + e_2 t - e_3 = (t-x)(t-y)(t-z).
	\]
	Since $e_3 = xyz \neq 0$ we may divide by any power of $x$, $y$, or $z$;
	this allows the recurrence to be run backwards, and hence extended to all integers $n \in \mathbb{Z}$.
	
	\subsection{Cyclic sum notation}
	
	Many expressions in this paper are cyclic in $x, y, z$.  We write
	\[
	\cyc{g(x)} \;:=\; g(x) + g(y) + g(z)
	\]
	for the cyclic sum of any expression $g$, where the substitution rule is $x \to y \to z \to x$.  For example,
	$e_1 = \cyc{x}$, $e_2 = \cyc{xy}$, $e_3 = xyz$, and in particular:
	\[
	\cyc{x^n(y-z)} \;=\; x^n(y-z) + y^n(z-x) + z^n(x-y).
	\]
	This notation eliminates the need to write all three terms of any
	cyclic identity.
	
	% -----------------------------------------------------------------------
	\section{Weighted Power Sums}
	
	\begin{definition}
		For any integer $n$ and a weight triple $(a, b, c)$, define
		\[
		f_n(a,b,c) = a x^n + b y^n + c z^n = \cyc{a\, x^n},
		\]
		where in the cyclic notation it is understood that the weights cycle
		as $(a,b,c) \to (b,c,a)$ alongside $(x,y,z) \to (y,z,x)$.
	\end{definition}
	
	The special case of immediate interest is:
	\[
	s_n \;:=\; f_n(1,1,1) \;=\; x^n + y^n + z^n \;=\; \cyc{x^n}.
	\]
	
	The key properties of $f_n$ are linearity in the weights and the universal
	recurrence inherited from $P(t)$.
	
	\begin{theorem}[Recurrence]
		\label{thm:recurrence}
		For any fixed weights $(a,b,c)$ and any integer $n$,
		\[
		f_{n+3}(a,b,c) = e_1 \cdot f_{n+2}(a,b,c) - e_2 \cdot f_{n+1}(a,b,c) + e_3 \cdot f_n(a,b,c).
		\]
	\end{theorem}
	\begin{proof}
		Each of $x,y,z$ satisfies $t^3 = e_1 t^2 - e_2 t + e_3$.  Multiply by
		$t^n$, weight by $a, b, c$ respectively, and sum.
	\end{proof}
	
	Because $e_3 \neq 0$ the recurrence can also be run backwards, making
	$f_n(a,b,c)$ well-defined for all $n \in \mathbb{Z}$.
	
	% -----------------------------------------------------------------------
	\section{The Basis $K$, $L$, $M$}
	
	Any sequence satisfying the recurrence of Theorem~\ref{thm:recurrence}
	is determined by its values at three consecutive integers.  We single
	out three basis sequences whose values at $n = 0, 1, 2$ form the columns of the identity matrix.
	
	\subsection{Definition via the alternating difference weights}
	
	The triple of \emph{difference weights}
	\[
	(w_x,\, w_y,\, w_z) \;:=\; (y-z,\; z-x,\; x-y)
	\]
	satisfies $w_x + w_y + w_z = 0$ and is nonzero since $x,y,z$ are
	distinct.  A short computation gives
	\[
	f_0(w_x,w_y,w_z) = 0, \quad f_1(w_x,w_y,w_z) = 0, \quad
	f_2(w_x,w_y,w_z) = \cyc{x^2(y-z)} =: \Delta,
	\]
	where we define $\Delta := -(y-z)(z-x)(x-y) \neq 0$, which is nonzero since $x,y,z$ are distinct.  (The $f_2$ identity follows by direct expansion.)
	
	\begin{definition}
		\label{def:K}
		For any integer $n$, define
		\[
		K(n) \;=\; \frac{f_n(w_x,\,w_y,\,w_z)}{\Delta}
		\;=\; \frac{\cyc{x^n(y-z)}}{\cyc{x^2(y-z)}}.
		\]
	\end{definition}
	
	We define two companion sequences so that $L$ and $M$ complete the standard basis with the correct initial values:
	
	\begin{definition}
		\label{def:LM}
		\[
		L(n) = K(n+1) - e_1 \cdot K(n), \qquad M(n) = e_3 \cdot K(n-1).
		\]
	\end{definition}
	
	The initial values follow immediately:
	
	\[
	\renewcommand{\arraystretch}{1.3}
	\begin{array}{c|ccc}
		n & 0 & 1 & 2 \\\hline
		K(n) & 0 & 0 & 1 \\
		L(n) & 0 & 1 & 0 \\
		M(n) & 1 & 0 & 0
	\end{array}
	\]
	
	The triple $(K, L, M)$ forms the \emph{standard basis} for all integer
	sequences satisfying the recurrence: the columns at $n=0,1,2$ are
	exactly the identity matrix.  All three sequences satisfy the same
	recurrence as $f_n$, since $K$ inherits it from $f_n$, and $L$, $M$
	are linear combinations of shifts of $K$.
	
	% -----------------------------------------------------------------------
	\section{The Decomposition Theorem}
	\label{sec:decomposition}
	
	The following is the central engine of the paper.
	
	\begin{theorem}[Decomposition]
		\label{thm:decomposition}
		For any weights $(a,b,c)$ and any integer $n$,
		\[
		f_n(a,b,c) \;=\; K(n) \cdot f_2(a,b,c) \;+\; L(n) \cdot f_1(a,b,c) \;+\; M(n) \cdot f_0(a,b,c).
		\]
	\end{theorem}
	\begin{proof}
		Both sides satisfy the same recurrence.  They agree at $n=0,1,2$ by
		the initial values of $K$, $L$, $M$.  Uniqueness of recurrence
		solutions gives equality for all $n$.
	\end{proof}
	
	\begin{corollary}[Shift invariance]
		\label{cor:shift}
		For any weights $(a,b,c)$ and any integers $n$, $j$,
		\[
		f_{n+j}(a,b,c) \;=\; K(n) \cdot f_{j+2}(a,b,c) \;+\; L(n) \cdot f_{j+1}(a,b,c) \;+\; M(n) \cdot f_j(a,b,c).
		\]
	\end{corollary}
	\begin{proof}
		Fix $j$.  The shifted sequence $h_n = f_{n+j}(a,b,c)$ satisfies the
		same recurrence, with $h_0 = f_j$, $h_1 = f_{j+1}$, $h_2 = f_{j+2}$.
		Apply Theorem~\ref{thm:decomposition} to $h_n$.
	\end{proof}
	
	The Decomposition Theorem says: \emph{to evaluate any weighted power sum
		at index $n$, it suffices to know it at three consecutive indices — and
		the coefficients are always $K(n)$, $L(n)$, $M(n)$, regardless of the
		weights}.
	
	\subsection{Addition formula for $K$}
	
	Since $K(n) = f_n(w_x,w_y,w_z)/\Delta$, applying the Shift Corollary to $K$ directly yields an addition formula that will be used in the divisibility arguments of Section~\ref{sec:flt}.
	
	\begin{lemma}[Addition formula]
		\label{lem:addition}
		For all integers $m$ and $n$,
		\[
		K(n+m) \;=\; K(n) \cdot K(m+2) + L(n) \cdot K(m+1) + M(n) \cdot K(m).
		\]
	\end{lemma}
	\begin{proof}
		Fix $n$ and consider both sides as functions of $m$.  The left-hand side $K(n+m)$ satisfies the recurrence in $m$ because $K$ does.  The right-hand side is a linear combination (with fixed $n$-dependent coefficients) of the sequences $m \mapsto K(m+2)$, $m \mapsto K(m+1)$, $m \mapsto K(m)$, each of which satisfies the same recurrence; hence so does their combination.  At
		$m = 0, 1, 2$ the right-hand side evaluates, using $K(0)=K(1)=0$ and
		$K(2)=1$, to
		\begin{align*}
			m=0: &\quad K(n)\cdot 1 + L(n)\cdot 0 + M(n)\cdot 0 = K(n),\\
			m=1: &\quad K(n)\cdot K(3) + L(n)\cdot 1 + M(n)\cdot 0 = e_1 K(n) + L(n) = K(n+1),\\
			m=2: &\quad K(n)\cdot K(4) + L(n)\cdot K(3) + M(n)\cdot 1,
		\end{align*}
		which equals $K(n+2)$ by the Decomposition Theorem applied to $K$.
		Uniqueness of recurrence solutions gives equality for all $m$.
	\end{proof}
	
	% -----------------------------------------------------------------------
	\section{Applications and Key Lemmas}
	\label{sec:applications}
	
	\subsection{Power sums in the $K$-basis}
	
	Apply the decomposition with weights $(1,1,1)$, noting
	$f_0(1,1,1)=3$, $f_1(1,1,1)=e_1$, $f_2(1,1,1)=e_1^2-2e_2$:
	\[
	s_n \;=\; K(n) \cdot (e_1^2-2e_2) \;+\; L(n) \cdot e_1 \;+\; 3 \cdot M(n).
	\]
	In particular, $s_0=3$, $s_1=e_1$, $s_2=e_1^2-2e_2$ match the standard values, and equality for all $n$ follows.  Expanding the
	definitions yields the key formula:
	
	\begin{lemma}
		\label{lem:sn-formula}
		$s_n \;=\; e_1 K(n+1) - 2e_2 K(n) + 3e_3 K(n-1)$ for all integers $n$.
	\end{lemma}
	\begin{proof}
		Apply the Decomposition Theorem with weights $(1,1,1)$, substituting
		$f_0(1,1,1)=3$, $f_1(1,1,1)=e_1$, $f_2(1,1,1)=e_1^2-2e_2$ and the
		definitions $L(n) = K(n+1)-e_1 K(n)$, $M(n) = e_3 K(n-1)$:
		\[
		s_n = K(n)(e_1^2-2e_2) + L(n) \cdot e_1 + 3 \cdot M(n)
		= K(n)(e_1^2-2e_2) + (K(n+1)-e_1 K(n))e_1 + 3e_3 K(n-1).
		\]
		Expanding and cancelling the $e_1^2 K(n)$ terms gives the result.
	\end{proof}
	
	\begin{theorem}
		\label{thm:sn-basis}
		$s_n \;=\; K(n+2) + L(n+1) + M(n)$ for all integers $n$.
	\end{theorem}
	\begin{proof}
		By Lemma~\ref{lem:sn-formula}, $s_n = e_1 K(n+1) - 2e_2 K(n) + 3e_3 K(n-1)$.
		Substituting $K(n+2) = e_1 K(n+1)-e_2 K(n)+e_3 K(n-1)$,
		$L(n+1) = K(n+2)-e_1 K(n+1)$, and $M(n) = e_3 K(n-1)$, one verifies
		that $K(n+2)+L(n+1)+M(n) = e_1 K(n+1) - 2e_2 K(n) + 3e_3 K(n-1) = s_n$.
	\end{proof}

	\section{Power Substitution Identities}
	\label{sec:substitutions}
	
	\subsection{The generalised function $K_n(a,b,c)$}
	
	The formula in Definition~\ref{def:K} makes sense for any distinct
	nonzero triple $(a,b,c)$.
	
	\begin{definition}
		\label{def:Kgen}
		For distinct nonzero $a,b,c$ and any integer $n$,
		\[
		K_n(a,b,c) \;=\; \frac{\cyc{a^n(b-c)}}{\cyc{a^2(b-c)}}
		\;=\; \frac{a^n(b-c)+b^n(c-a)+c^n(a-b)}{-(b-c)(c-a)(a-b)}.
		\]
	\end{definition}
	
	So $K_n(x,y,z) = K(n)$.  The function $K_n(a,b,c)$ satisfies the
	recurrence with symmetric polynomials $\varepsilon_i$ of $(a,b,c)$ in
	place of $e_i$, and has initial values $K_0=K_1=0$, $K_2=1$ for any
	admissible triple.
	
	The utility of this definition is that \emph{evaluating $K_n$ at a
		specific triple reduces, via the Decomposition Theorem, to a clean
		identity in $K$, $L$, $M$}.
	
	\subsection{General power substitution}
	
	For an integer $k \geq 1$, write the \emph{$k$-th difference discriminant} as
	\[
	\Delta_k \;=\; -(y^k-z^k)(z^k-x^k)(x^k-y^k),
	\]
	consistently with the sign convention $\Delta = \Delta_1 = -(y-z)(z-x)(x-y)$.
	The ratio $\Delta_k/\Delta$ is a symmetric polynomial in $x,y,z$; for
	example $\Delta_2/\Delta = (x+y)(y+z)(z+x) = -(e_3 - e_1 e_2)$.
	
	\begin{theorem}[General power substitution]
		\label{thm:power}
		For all integers $k \geq 1$ and $n$,
		\[
		\frac{\Delta_k}{\Delta}\, K_n(x^k,y^k,z^k)
		\;=\;
		K(kn)\,K(k+1) - K(kn+1)\,K(k).
		\]
	\end{theorem}
	\begin{proof}
		By definition, the numerator of $K_n(x^k,y^k,z^k)$ is
		\[
		\cyc{(x^k)^n(y^k-z^k)} \;=\; f_{kn}(y^k-z^k,\, z^k-x^k,\, x^k-y^k),
		\]
		since $(x^k)^n \cdot (y^k - z^k)$ is the $x$-term of $f_{kn}$ with weights
		$(y^k-z^k, z^k-x^k, x^k-y^k)$.  Denote these weights $(p,q,r)$.
		We evaluate $f_0, f_1, f_2$ at $(p,q,r)$:
		\begin{itemize}
			\item $f_0(p,q,r) = 0$, since $p+q+r = 0$.
			\item $f_1(p,q,r) = \cyc{x(y^k-z^k)}$.  Expanding the cyclic sum
			and collecting the $x^k$ terms (each $x \cdot y^k$ pairs with
			$-y \cdot x^k$ from the next term):
			$\cyc{x(y^k-z^k)} = \cyc{x^k(z-y)} = -\cyc{x^k(y-z)} = -K(k) \cdot \Delta$.
			\item $f_2(p,q,r) = \cyc{x^2(y^k-z^k)} = -\cyc{x^k(y^2-z^2)}$.
			Writing $y^2-z^2 = (y-z)(y+z) = -w_x(y+z)$ and noting that $e_1 - x = y+z$ (and cyclically), we get
			$\cyc{x^k w_x(y+z)} = \cyc{x^k w_x(e_1-x)} = e_1 \cdot f_k(w_x,w_y,w_z) - f_{k+1}(w_x,w_y,w_z)$.
			Hence $\cyc{x^k(y^2-z^2)} = e_1 \cdot f_k(w_x,w_y,w_z) - f_{k+1}(w_x,w_y,w_z)
			= (e_1 K(k) - K(k+1))\Delta = -L(k) \cdot \Delta$,
			so $f_2(p,q,r) = L(k) \cdot \Delta$.
		\end{itemize}
		Applying the Decomposition Theorem (with $kn$ in place of $n$):
		\[
		f_{kn}(p,q,r) \;=\; K(kn) \cdot L(k) \cdot \Delta \;+\; L(kn) \cdot (-K(k)) \cdot \Delta,
		\]
		and since $f_{kn}(p,q,r) = K_n(x^k,y^k,z^k)\cdot\Delta_k$, dividing by $\Delta$:
		\[
		\frac{\Delta_k}{\Delta}\,K_n(x^k,y^k,z^k) = K(kn) \cdot L(k) - L(kn) \cdot K(k).
		\]
		Substituting $L(m) = K(m+1) - e_1 K(m)$ into both $L(k)$ and $L(kn)$ and
		expanding, the $e_1$ terms cancel, leaving $K(kn) \cdot K(k+1) - K(kn+1) \cdot K(k)$.
	\end{proof}
	
	\subsection*{Special case $n=2$}
	
	Since $K_2(a,b,c) = 1$ for any triple, the theorem gives the closed formula
	\[
	\frac{\Delta_k}{\Delta}
	\;=\;
	K(2k)\,K(k+1) - K(2k+1)\,K(k).
	\]
	
	\subsection*{Special case $n=3$}
	
	For any triple $(a,b,c)$ one has $K_3(a,b,c) = a+b+c$ directly from the
	closed form.  To see this, note that $a^3(b-c) + b^3(c-a) + c^3(a-b)$
	$= (a+b+c)(a^2(b-c)+b^2(c-a)+c^2(a-b))$, i.e.\ the numerator $\cyc{a^3(b-c)}$ factors as
	$(a+b+c)\cdot\cyc{a^2(b-c)}$.  Applied to $(x^k,y^k,z^k)$ this gives
	$K_3(x^k,y^k,z^k) = s_k$, and the theorem yields
	\[
	\frac{\Delta_k}{\Delta}\, s_k
	\;=\;
	K(3k)\,K(k+1) - K(3k+1)\,K(k).
	\]
	
	\subsection{The squaring substitution: $(a,b,c) = (x^2, y^2, z^2)$}
	
	The triple $(x^2,y^2,z^2)$ has symmetric polynomials
	\[
	\varepsilon_1 = e_1^2-2e_2, \qquad \varepsilon_2 = e_2^2-2e_1e_3, \qquad \varepsilon_3 = e_3^2.
	\]
	
	\begin{corollary}[Squaring-substitution identity]
		\label{thm:squaring}
		For all integers $n$,
		\[
		(e_3 - e_1 e_2)\cdot K_n(x^2,\,y^2,\,z^2) \;=\; L(2n).
		\]
	\end{corollary}
	\begin{proof}
		Set $k=2$ in Theorem~\ref{thm:power}.  The prefactor satisfies
		$\Delta_2/\Delta = (x+y)(y+z)(z+x) = -(e_3-e_1e_2)$, and the
		right-hand side becomes
		$K(2n)\,K(3) - K(2n+1)\,K(2) = e_1 K(2n) - K(2n+1) = -L(2n)$,
		since $K(2)=1$, $K(3)=e_1$, and $L(2n) = K(2n+1)-e_1 K(2n)$.
		Combining signs gives the stated identity.
	\end{proof}
	
	% -----------------------------------------------------------------------
	\section{Consequences of $s_n = 0$ over the reals}
	\label{sec:sn-zero}
	
	Throughout this section $x$, $y$, $z$ are distinct nonzero real (or
	complex) numbers satisfying $s_n = x^n + y^n + z^n = 0$ for a fixed
	integer $n \geq 3$.  No integrality of $x$, $y$, $z$ is assumed in this section.
	
	\subsection{The ratio identity}
	
	Setting $k = n$ and $m = 3$ in Theorem~\ref{thm:power} and using
	$K_3(x^n, y^n, z^n) = s_n = 0$ gives immediately:
	
	\begin{corollary}
		\label{cor:sn-ratio}
		If $s_n = 0$ then $K(3n) \cdot K(n+1) = K(3n+1) \cdot K(n)$.
	\end{corollary}
	\begin{proof}
		Set $k = n$ and use $K_3(x^n,y^n,z^n) = s_n = 0$ in Theorem~\ref{thm:power}; the right-hand side vanishes.
	\end{proof}
	
	In other words the two ratios $K(3n)/K(n)$ and $K(3n+1)/K(n+1)$ are
	equal (whenever the denominators are nonzero).  We write their common
	value $A = K(3n)/K(n)$.
	
	\subsection{The common ratio $A$}
	
	\begin{theorem}
		\label{thm:sn-ratio}
		Suppose $s_n = 0$ and $K(n) \neq 0$.  Let $A = K(3n)/K(n)$.  Then
		\[
		K(3n+1) = A\,K(n+1).
		\]
	\end{theorem}
	\begin{proof}
		The ratio identity (Corollary~\ref{cor:sn-ratio}) gives
		$K(3n) \cdot K(n+1) = K(3n+1) \cdot K(n)$.  Substituting $K(3n) = A \cdot K(n)$
		and dividing both sides by $K(n)$ yields the result.
	\end{proof}
	
	\subsection{The master identity}
	
	The two equalities above are the $j = 0$ and $j = 1$ cases of a
	uniform identity covering every offset $j$.
	
	\begin{theorem}[Master identity]
		\label{thm:master}
		Suppose $s_n = 0$ and $K(n) \neq 0$.  Let $A = K(3n)/K(n)$.  Then for all integers $j$,
		\[
		K(3n+j) \;=\; A \cdot K(n+j) \;+\; e_3^n \cdot K(j).
		\]
	\end{theorem}
	\begin{proof}
		Fix $n$.  The right-hand side, viewed as a function of $j$, is a sum
		of two sequences: $j \mapsto A \cdot K(n+j)$ (a scalar multiple of a shift
		of $K$) and $j \mapsto e_3^n \cdot K(j)$ (a scalar multiple of $K$ itself).
		Both satisfy the recurrence with coefficients $e_1, e_2, e_3$, and
		therefore so does their sum.  Since the left-hand side also satisfies
		the same recurrence, it suffices to check agreement at $j = 0, 1, 2$.
		
		\smallskip\noindent
		$j = 0$: $A \cdot K(n) + e_3^n \cdot 0 = A \cdot K(n) = K(3n)$. \checkmark
		
		\smallskip\noindent
		$j = 1$: $A \cdot K(n+1) + e_3^n \cdot 0 = A \cdot K(n+1) = K(3n+1)$,
		by Theorem~\ref{thm:sn-ratio}. \checkmark
		
		\smallskip\noindent
		$j = 2$: we need $K(3n+2) = A \cdot K(n+2) + e_3^n$.
		Apply the $s_n$ formula (Lemma~\ref{lem:sn-formula}) at index $3n$:
		\[
		s_{3n} = e_1 \cdot K(3n+1) - 2e_2 \cdot K(3n) + 3e_3 \cdot K(3n-1).
		\]
		Substituting $K(3n) = A\,K(n)$, $K(3n+1) = A\,K(n+1)$, and using
		$s_n = 0$ to replace $e_1 \cdot K(n+1) - 2e_2 \cdot K(n) = -3e_3 \cdot K(n-1)$:
		\[
		s_{3n} = 3e_3 \cdot \bigl(K(3n-1) - A \cdot K(n-1)\bigr).
		\]
		Since $x^n + y^n + z^n = 0$, we apply Newton's identity to the triple
		$(x^n, y^n, z^n)$ with elementary symmetric polynomials $\tilde{e}_1 = s_n = 0$,
		$\tilde{e}_2 = x^ny^n+y^nz^n+z^nx^n$, $\tilde{e}_3 = e_3^n$.  The Newton identity
		$p_3 = \tilde{e}_1 p_2 - \tilde{e}_2 p_1 + 3\tilde{e}_3$ gives
		$s_{3n} = 0 \cdot s_{2n} - \tilde{e}_2 \cdot 0 + 3e_3^n = 3e_3^n$, so
		$K(3n-1) = A \cdot K(n-1) + e_3^{n-1}$.  Now the recurrence at $3n+2$:
		\begin{align*}
			K(3n+2)
			&= e_1 \cdot A \cdot K(n+1) - e_2 \cdot A \cdot K(n)
			+ e_3 \cdot (A \cdot K(n-1) + e_3^{n-1}) \\
			&= A \cdot \bigl(e_1 \cdot K(n+1) - e_2 \cdot K(n) + e_3 \cdot K(n-1)\bigr) + e_3^n
			= A \cdot K(n+2) + e_3^n. \quad\checkmark
		\end{align*}
		Uniqueness of recurrence solutions gives equality for all $j$.
	\end{proof}
	
	The $j = -1$ instance of the master identity is worth recording explicitly,
	since it was already established inside the proof above:
	
	\begin{corollary}
		\label{cor:K3n-1}
		If $s_n = 0$ and $K(n) \neq 0$, then
		\[
		K(3n-1) \;=\; A \cdot K(n-1) \;+\; e_3^{n-1}.
		\]
	\end{corollary}
	\begin{proof}
		From the $s_{3n}$ calculation in the proof of Theorem~\ref{thm:master}:
		$s_{3n} = 3e_3 \cdot (K(3n-1) - A \cdot K(n-1))$ and $s_{3n} = 3e_3^n$,
		so $K(3n-1) - A \cdot K(n-1) = e_3^{n-1}$.
	\end{proof}

	\section{Constraints on a hypothetical counter-example to Fermat's Last Theorem}
	\label{sec:flt}
	
	\noindent\textit{Disclaimer.}
	Fermat's Last Theorem was proved by Wiles in 1995, and nothing here
	comes close to an independent proof --- nor is that the aim.  The goal
	is to see what arithmetic constraints the framework imposes on a
	hypothetical primitive solution $x^n + y^n + z^n = 0$ with
	$x, y, z \in \mathbb{Z}$.
	
	\medskip
	Throughout this section $x$, $y$, $z$ are pairwise coprime nonzero
	integers with $s_n = 0$ and $n \geq 3$ an odd prime.  All results of
	Section~\ref{sec:sn-zero} apply; in particular, $A$ and the master identity (Theorem~\ref{thm:master}) remain valid.  The additional content here is purely about divisibility.
	
	\subsection{Divisibility consequences of the master identity}
	
	Since $x$, $y$, $z$ are integers, every $K(m)$ is an integer.
	We now show $K(n) \mid K(3n)$, after which divisibility of $K(3n+1)$
	and the full master identity follow with no further calculation.
	
	\begin{theorem}
		\label{thm:flt-divisibility}
		If $x$, $y$, $z$ are coprime integers with $s_n = 0$, then
		$K(n) \mid K(3n)$, and consequently $K(n+1) \mid K(3n+1)$, with
		the same integer quotient $A = K(3n)/K(n)$.
	\end{theorem}
	\begin{proof}
		We show $K(3n) \equiv 0 \pmod{K(n)}$.  Applying the addition formula
		(Lemma~\ref{lem:addition}) with $m = n$ and $m=n+1$:
		\begin{align*}
			K(2n)   &\equiv L(n) \cdot K(n+1) &\mod K(n), \\
			K(2n+1) &\equiv L(n) \cdot K(n+2) + M(n) \cdot K(n+1) &\mod K(n).
		\end{align*}
		Applying the addition formula once more with $m = 2n$ and reducing modulo $K(n)$, and substituting
		$L(n) \equiv K(n+1)$, $M(n) \equiv e_3 \cdot K(n-1)$,
		and $K(n+2) \equiv e_1 \cdot K(n+1) + e_3 \cdot K(n-1)$, gives
		\begin{align*}
			K(3n) &\;\equiv\; L(n) \cdot K(2n+1) + M(n) \cdot K(2n) &\mod K(n),\\
			K(3n) &\;\equiv\; K(n+1)^2 \cdot \bigl(e_1 \cdot K(n+1) + 3e_3 \cdot K(n-1)\bigr) &\mod K(n).
		\end{align*}
		Since $s_n = 0$, Lemma~\ref{lem:sn-formula} gives
		$e_1 \cdot K(n+1) + 3e_3 \cdot K(n-1) = 2e_2 \cdot K(n) \equiv 0 \pmod{K(n)}$.
		Hence $K(3n) \equiv 0 \pmod{K(n)}$, so $A = K(3n)/K(n)$ is an integer.
		
		Since $A$ is an integer, Theorem~\ref{thm:sn-ratio} gives
		$K(3n+1) = A \cdot K(n+1)$, so $K(n+1) \mid K(3n+1)$ with the same
		quotient $A$.
	\end{proof}
	
	Since $A$ is an integer, the master identity now takes on its full arithmetic force:
	
	\begin{corollary}[Master identity over $\mathbb{Z}$]
		\label{cor:master-int}
		If $x$, $y$, $z$ are pairwise coprime nonzero integers with $s_n = 0$
		and $n \geq 3$ an odd prime, then all of $K(m)$, $e_3^n$, and
		$A = K(3n)/K(n)$ are integers, and for all integers $j$,
		\[
		K(3n+j) \;=\; A \cdot K(n+j) \;+\; e_3^n \cdot K(j).
		\]
		In particular, $K(n) \mid K(3n+j) - e_3^n \cdot K(j)$ for all $j$.
	\end{corollary}
	\begin{proof}
		Integrality of $K(m)$ follows since $x,y,z \in \mathbb{Z}$.
		Integrality of $A$ is Theorem~\ref{thm:flt-divisibility}.
		The identity is Theorem~\ref{thm:master}; divisibility follows immediately.
	\end{proof}
	
	\begin{theorem}
		\label{thm:K-mod-A}
		Under the standing hypotheses, for all integers $m \geq 1$ and all $k$,
		\[
		K(3nm + k) \;\equiv\; e_3^{nm} \cdot K(k) \pmod{A}.
		\]
		In particular:
		\begin{itemize}
			\item $A \mid K(3nm)$ and $A \mid K(3nm+1)$ for all $m \geq 1$, since $K(0) = K(1) = 0$.
			\item $K(3nm+2) \equiv e_3^{nm} \pmod{A}$ for all $m \geq 1$, since $K(2) = 1$.
		\end{itemize}
	\end{theorem}
	\begin{proof}
		Induct on $m$.  For $m=1$, the master identity
		(Corollary~\ref{cor:master-int}) with $j = k$ gives
		\[
		K(3n+k) = A \cdot K(n+k) + e_3^n \cdot K(k) \;\equiv\; e_3^n \cdot K(k) \pmod{A}.
		\]
		For the inductive step, apply the master identity with $j = 3n(m-1)+k$:
		\[
		K(3nm+k) = A \cdot K\bigl(3n(m-1)+n+k\bigr) + e_3^n \cdot K\bigl(3n(m-1)+k\bigr).
		\]
		Reducing mod $A$ and applying the inductive hypothesis to the second term:
		\[
		K(3nm+k) \;\equiv\; e_3^n \cdot e_3^{n(m-1)} \cdot K(k) = e_3^{nm} \cdot K(k) \pmod{A}. \qedhere
		\]
	\end{proof}
	
	\begin{corollary}
		\label{cor:s3nr-mod-A}
		Under the standing hypotheses, for all integers $r$,
		\[
		s_{3n+r} \;\equiv\; e_3^n \cdot s_r \pmod{A}.
		\]
		In particular, $s_{3nm+r} \equiv e_3^{nm} \cdot s_r \pmod{A}$ for all $m \geq 1$,
		by Theorem~\ref{thm:K-mod-A} applied via the same reduction.
	\end{corollary}
	\begin{proof}
		Apply the shift corollary (Corollary~\ref{cor:shift}) with weights $(1,1,1)$
		and shift $j = r$, reading off at index $3n$:
		\[
		s_{3n+r} \;=\; K(3n)\,s_{r+2} + L(3n)\,s_{r+1} + M(3n)\,s_r.
		\]
		Since $K(3n) \equiv 0$ and $K(3n+1) \equiv 0 \pmod{A}$,
		we have $L(3n) = K(3n+1) - e_1\,K(3n) \equiv 0 \pmod{A}$.
		For $M(3n) = e_3\,K(3n-1)$, Corollary~\ref{cor:K3n-1} gives
		$K(3n-1) = A \cdot K(n-1) + e_3^{n-1}$, so $M(3n) \equiv e_3^n \pmod{A}$.
		Hence $s_{3n+r} \equiv e_3^n \cdot s_r \pmod{A}$.
	\end{proof}
	
	\begin{lemma}
		\label{lem:etilde-coprime}
		$\gcd(\tilde{e}_2,\, e_3) = 1$, where
		$\tilde{e}_2 = x^ny^n + y^nz^n + z^nx^n$ and $e_3 = xyz$.
	\end{lemma}
	\begin{proof}
		Let $p$ be any prime dividing $e_3 = xyz$.  Since $x$, $y$, $z$ are
		pairwise coprime, $p$ divides exactly one of them, say $x$.  Then
		$p \mid x^ny^n$ and $p \mid z^nx^n$, but $p \nmid y^nz^n$, so
		$p \nmid \tilde{e}_2 = x^ny^n + y^nz^n + z^nx^n$.
		Since no prime divides both $\tilde{e}_2$ and $e_3$, they are coprime.
	\end{proof}
	
	\begin{corollary}
		\label{cor:gcd-A-e3}
		Under the standing hypotheses, $A \mid s_{4n}$ and $\gcd(A,\, e_3) \mid 2$.
	\end{corollary}
	\begin{proof}
		Setting $r = n$ in Corollary~\ref{cor:s3nr-mod-A} and using $s_n = 0$:
		\[
		s_{4n} \;\equiv\; e_3^n\,s_n = 0 \pmod{A},
		\]
		so $A \mid s_{4n}$.  To evaluate $s_{4n}$, apply Newton's identities to the triple $(x^n,y^n,z^n)$
		with $\tilde{e}_1 = s_n = 0$, $\tilde{e}_2 = x^ny^n+y^nz^n+z^nx^n$, $\tilde{e}_3 = e_3^n$.
		First, $s_{2n} = \tilde{e}_1^2 - 2\tilde{e}_2 = -2\tilde{e}_2$.
		Then $s_{4n} = \tilde{e}_1 s_{3n} - \tilde{e}_2 s_{2n} + \tilde{e}_3 s_n
		= 0 - \tilde{e}_2(-2\tilde{e}_2) + 0 = 2\tilde{e}_2^2$, so
		$A \mid 2\tilde{e}_2^2$.  Since $\gcd(A, e_3)$ divides both
		$A$ and $e_3 = xyz$, it divides
		\[
		\gcd(2\tilde{e}_2^2,\, e_3).
		\]
		By Lemma~\ref{lem:etilde-coprime}, $\gcd(\tilde{e}_2, e_3) = 1$,
		so $\gcd(\tilde{e}_2^2, e_3) = 1$, and therefore
		$\gcd(2\tilde{e}_2^2, e_3) = \gcd(2, e_3) \mid 2$.
		Hence $\gcd(A,\, e_3) \mid 2$.
	\end{proof}
