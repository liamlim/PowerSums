	\section{Setup and Conventions}
	\label{sec:setup}
	% ---------------------------------------------------------------------------
	
	Throughout this section, let $(x,y,z)\in\mathbb{R}^3$ be a fixed triplet such that:
	\begin{itemize}
		\item $xyz \neq 0$,
		\item $x \neq y, y \neq z, \text{ and } z \neq x$.
	\end{itemize}
	
	\begin{definition}
		We define the elementary symmetric polynomials $e_1, e_2, e_3$ as
		\begin{align*}
			e_1 &= x + y + z = \cycsum{x}, \\
			e_2 &= xy + yz + zx = \cycsum{xy}, \\
			e_3 &= xyz. 
		\end{align*}
	\end{definition}
	
	\begin{definition}
		For any $n \in \mathbb{Z}$ and weights $(a, b, c) \in \mathbb{R}^3$, we define the weighted power sum $f_n(a, b, c)$ by
		\[ f_n(a, b, c) = ax^n + by^n + cz^n. \]
	\end{definition}
	
	\begin{definition}
		For any $n \in \mathbb{Z}$, let $s_n$ denote the absolute power sum
		\[ s_n = f_n(1, 1, 1) = x^n + y^n + z^n = \cycsum{x^n}. \]
	\end{definition}
	
	\begin{definition}
		We define the weights $w_x, w_y, w_z$ as
		\begin{align*}
			w_x &= y - z, \\
			w_y &= z - x, \\
			w_z &= x - y.
		\end{align*}
	\end{definition}
	
	\begin{theorem}[Vanishing Sums]
		\label{thm:vanishing}
		The following identities hold for the weighted power sums:
		\begin{enumerate}
			\item $f_0(w_x, w_y, w_z) = 0$
			\item $f_1(w_x, w_y, w_z) = 0$
		\end{enumerate}
	\end{theorem}
	
	\begin{proof}
		For the first identity, by the definition of $s_0$, we have:
		\[
		f_0(w_x, w_y, w_z) = w_x x^0 + w_y y^0 + w_z z^0 = \cycsum{w_x}.
		\]
		Substituting the definitions of the weights:
		\[
		(y - z) + (z - x) + (x - y) = 0.
		\]
		
		For the second identity, the definition of $f_1$ gives:
		\[
		f_1(w_x, w_y, w_z) = w_x x^1 + w_y y^1 + w_z z^1 = \cycsum{x(y-z)}.
		\]
		Expanding the terms:
		\[
		xy - xz + yz - yx + zx - zy = 0.
		\]
		This completes the proof.
	\end{proof}
	
	\begin{definition}
		We define $\Delta$ as the weighted power sum of degree 2:
		\[ \Delta = f_2(w_x, w_y, w_z) = w_x x^2 + w_y y^2 + w_z z^2 = \cycsum{w_x x^2}. \]
	\end{definition}
	
	\begin{lemma}
		\label{lem:delta_factored}
		$\Delta = -w_x w_y w_z.$
		In particular, $\Delta \neq 0$ since $x, y, z$ are pairwise distinct.
	\end{lemma}
	
	\begin{proof}
		Expanding $-w_x w_y w_z = -(y-z)(z-x)(x-y)$:
		\begin{align*}
			-(y-z)(z-x)(x-y)
			&= -(y-z)\bigl(zx - zy - x^2 + xy\bigr) \\
			&= -(zxy - zy^2 - x^2y + xy^2 - z^2x + z^2y + x^2z - xyz) \\
			&= x^2(y-z) - y^2(x-z) + z^2(x-y) \\
			&= w_x x^2 + w_y y^2 + w_z z^2 = \Delta. \qedhere
		\end{align*}
	\end{proof}
	
	\begin{definition}
		For each $n \in \mathbb{Z}$, we define the normalized power sum $K(n)$ by
		\[ K(n) = \frac{f_n(w_x, w_y, w_z)}{\Delta}. \]
	\end{definition}
	
	% ---------------------------------------------------------------------------
	\section{Universal Recurrence}
	\label{sec:recurrence}
	% ---------------------------------------------------------------------------
	
	\begin{definition}
		We define the \emph{characteristic polynomial} $P(t)$ associated with the triplet $(x, y, z)$ as:
		\[ P(t) = (t-x)(t-y)(t-z) = t^3 - e_1 t^2 + e_2 t - e_3. \]
	\end{definition}
	
	\begin{theorem}[Universal Recurrence]
		\label{thm:universal_recurrence}
		For any fixed weights $(a, b, c)$ and any integer $n$, the weighted power sum $f_n(a, b, c)$ satisfies the linear recurrence:
		\[ f_{n+3} = e_1 f_{n+2} - e_2 f_{n+1} + e_3 f_n. \]
	\end{theorem}
	
	\begin{proof}
		By definition, $x, y, \text{ and } z$ are the roots of $P(t) = 0$. Therefore, for each $t \in \{x, y, z\}$, we have:
		\[ t^3 - e_1 t^2 + e_2 t - e_3 = 0 \implies t^{n+3} = e_1 t^{n+2} - e_2 t^{n+1} + e_3 t^n. \]
		We multiply this equation by the respective weights $a, b, \text{ and } c$:
		\begin{align*}
			a x^{n+3} &= a(e_1 x^{n+2} - e_2 x^{n+1} + e_3 x^n) \\
			b y^{n+3} &= b(e_1 y^{n+2} - e_2 y^{n+1} + e_3 y^n) \\
			c z^{n+3} &= c(e_1 z^{n+2} - e_2 z^{n+1} + e_3 z^n)
		\end{align*}
		Summing these three equations yields:
		\[ \cycsum{ax^{n+3}} = e_1 \cycsum{ax^{n+2}} - e_2 \cycsum{ax^{n+1}} + e_3 \cycsum{ax^n}. \]
		By the definition of $f_n$, this is exactly $f_{n+3} = e_1 f_{n+2} - e_2 f_{n+1} + e_3 f_n$.
	\end{proof}
	
	\begin{corollary}[Recurrence of $K$]
		\label{cor:recurrence_K}
		The normalized sequence $K(n)$ satisfies the same recurrence:
		\[ K(n+3) = e_1 K(n+2) - e_2 K(n+1) + e_3 K(n). \]
	\end{corollary}
	
	\begin{proof}
		Recall that $K(n) = f_n(w_x, w_y, w_z) / \Delta$. Since Theorem~\ref{thm:universal_recurrence} is linear in the weights $(a, b, c)$ and $\Delta \neq 0$ is a constant, dividing throughout by $\Delta$ gives the same recurrence for $K(n)$.
	\end{proof}
	
	\subsection{Definition L, M}
	\begin{definition}
		We define the companion sequences $L(n)$ and $M(n)$ for all $n \in \mathbb{Z}$ by:
		\begin{align*}
			L(n) &= K(n+1) - e_1 K(n), \\
			M(n) &= e_3 K(n-1).
		\end{align*}
	\end{definition}
	
	\begin{corollary}[Recurrence of $L$ and $M$]
		\label{cor:recurrence_LM}
		Both companion sequences satisfy the same recurrence:
		\[ L(n+3) = e_1 L(n+2) - e_2 L(n+1) + e_3 L(n), \]
		\[ M(n+3) = e_1 M(n+2) - e_2 M(n+1) + e_3 M(n). \]
	\end{corollary}
	
	\begin{proof}
		Since $L(n) = K(n+1) - e_1 K(n)$ is a linear combination of shifted copies of $K$, and each shifted copy satisfies the Universal Recurrence, so does $L(n)$. Similarly, $M(n) = e_3 K(n-1)$ is a constant scalar multiple of a shifted copy of $K$, and therefore satisfies the same recurrence.
	\end{proof}
	
	\begin{corollary}[Alternative expression for $M$]
		\label{m_by_kl}
		For all $n \in \mathbb{Z}$,
		\[ M(n) = L(n+1) + e_2 K(n). \]
	\end{corollary}
	
	\begin{proof}
		Expanding the right-hand side using the definitions of $L$ and $K$:
		\[ L(n+1) + e_2 K(n) = K(n+2) - e_1 K(n+1) + e_2 K(n). \]
		Shifting the Universal Recurrence $K(n+3) = e_1 K(n+2) - e_2 K(n+1) + e_3 K(n)$ by $n \mapsto n-1$ and rearranging gives:
		\[ e_3 K(n-1) = K(n+2) - e_1 K(n+1) + e_2 K(n). \]
		The left-hand side is exactly $M(n)$ by definition.
	\end{proof}
	
	% ---------------------------------------------------------------------------
	\subsection{Initial Values for K, L, M}
	\label{sec:initial}
	% ---------------------------------------------------------------------------
	
	We now compute the values of $K$, $L$, and $M$ at indices $n = 0, 1, 2$.
	
	\begin{theorem}[Initial Values]
		\label{thm:initial_values}
		The sequences $K$, $L$, $M$ take the following values:
		\begin{center}
			\begin{tabular}{c|ccc}
				$n$ & $K(n)$ & $L(n)$ & $M(n)$ \\
				\hline
				$0$ & $0$ & $0$ & $1$ \\
				$1$ & $0$ & $1$ & $0$ \\
				$2$ & $1$ & $0$ & $0$ \\
			\end{tabular}
		\end{center}
	\end{theorem}
	
	\begin{proof}
		\textit{Values of $K$.}
		$K(0) = 0$ and $K(1) = 0$ follow directly from Theorem~\ref{thm:vanishing},
		since $f_0(w_x,w_y,w_z) = f_1(w_x,w_y,w_z) = 0$.
		For $K(2)$, the numerator is $f_2(w_x,w_y,w_z) = \Delta$ by definition, so $K(2) = 1$.
		
		\textit{Values of $L$:}
		Using $L(n) = K(n+1) - e_1 K(n)$:
		\begin{align*}
			L(0) &= K(1) - e_1 K(0) = 0, \\
			L(1) &= K(2) - e_1 K(1) = 1, \\
			L(2) &= K(3) - e_1 K(2).
		\end{align*}
		Applying the recurrence at $n=0$ gives $K(3) = e_1 K(2) - e_2 K(1) + e_3 K(0) = e_1$,
		so $L(2) = e_1 - e_1 = 0$.
		
		\textit{Values of $M$:}
		Using $M(n) = e_3 K(n-1)$:
		\begin{align*}
			M(1) &= e_3 K(0) = 0, \\
			M(2) &= e_3 K(1) = 0.
		\end{align*}
		For $M(0)$, we use Corollary~\ref{m_by_kl}: $M(0) = L(1) + e_2 K(0) = 1 + 0 = 1$.
	\end{proof}

	\section{Decomposition Theorem}
	\label{sec:decomposition}
	% ---------------------------------------------------------------------------
	
	\begin{theorem}[Decomposition]
		\label{thm:decomposition}
		For any weights $(a, b, c)$ and any integers $n, m$,
		\[ f_{n+m}(a,b,c) = K(n)\,f_{m+2}(a,b,c) + L(n)\,f_{m+1}(a,b,c) + M(n)\,f_m(a,b,c). \]
	\end{theorem}
	
	\begin{proof}
		Fix $m$ and $(a,b,c)$, and regard both sides as sequences in $n$. The left-hand
		side $f_{n+m}$ satisfies the Universal Recurrence in $n$ by Theorem~\ref{thm:universal_recurrence}.
		The right-hand side is a linear combination of $K(n)$, $L(n)$, $M(n)$ with
		constant coefficients $f_{m+2}$, $f_{m+1}$, $f_m$; since each of $K$, $L$, $M$
		satisfies the same recurrence by Corollaries~\ref{cor:recurrence_K} and~\ref{cor:recurrence_LM},
		so does the right-hand side.
		
		It therefore suffices to check equality at $n = 0, 1, 2$, which follows immediately
		from Theorem~\ref{thm:initial_values}:
		\begin{align*}
			n=0: &\quad K(0)\,f_{m+2} + L(0)\,f_{m+1} + M(0)\,f_m = f_m = f_{0+m}, \\
			n=1: &\quad K(1)\,f_{m+2} + L(1)\,f_{m+1} + M(1)\,f_m = f_{m+1} = f_{1+m}, \\
			n=2: &\quad K(2)\,f_{m+2} + L(2)\,f_{m+1} + M(2)\,f_m = f_{m+2} = f_{2+m}.
		\end{align*}
		Since the recurrence uniquely determines a sequence from any three consecutive values, the two sides agree for all $n \in \mathbb{Z}$.
	\end{proof}
	
	\begin{corollary}[Decomposition at $m = 0$]
		\label{cor:decomp_m0}
		For any weights $(a, b, c)$ and any integer $n$,
		\[ f_n(a,b,c) = K(n)\,f_2(a,b,c) + L(n)\,f_1(a,b,c) + M(n)\,f_0(a,b,c). \]
		In particular, every weighted power sum is a linear combination of $f_0$, $f_1$,
		$f_2$ with coefficients drawn from $K$, $L$, $M$.
	\end{corollary}
	
	\begin{corollary}[Decomposition for $K$]
		\label{cor:decomp_K}
		For any integers $n, m$,
		\[ K(n+m) = K(n)\,K(m+2) + L(n)\,K(m+1) + M(n)\,K(m). \]
	\end{corollary}
	
	\begin{proof}
		Fix $m$ and regard both sides as sequences in $n$. Both satisfy the Universal
		Recurrence in $n$: the left-hand side by Corollary~\ref{cor:recurrence_K}, and the right-hand side
		since $K$, $L$, $M$ do by Corollaries~\ref{cor:recurrence_K} and~\ref{cor:recurrence_LM}.
		Equality at $n = 0, 1, 2$ follows from Theorem~\ref{thm:initial_values}:
		\begin{align*}
			n=0: &\quad K(0)\,K(m+2) + L(0)\,K(m+1) + M(0)\,K(m) = K(m) = K(0+m), \\
			n=1: &\quad K(1)\,K(m+2) + L(1)\,K(m+1) + M(1)\,K(m) = K(m+1) = K(1+m), \\
			n=2: &\quad K(2)\,K(m+2) + L(2)\,K(m+1) + M(2)\,K(m) = K(m+2) = K(2+m).
		\end{align*}
		Since the recurrence uniquely determines a sequence from three consecutive values, the two sides agree for all $n \in \mathbb{Z}$.
	\end{proof}
	
	\begin{corollary}[$s_n$ in terms of $K$]
		\label{cor:sn_K}
		For any integer $n$,
		\[ s_n = 3K(n+2) - 2e_1 K(n+1) + e_2 K(n). \]
	\end{corollary}
	
	\begin{proof}
		Since $s_n = f_n(1,1,1)$, Corollary~\ref{cor:decomp_m0} gives:
		\[ s_n = K(n)\,s_2 + L(n)\,s_1 + M(n)\,s_0. \]
		Substituting $s_0 = 3$, $s_1 = e_1$, $s_2 = e_1^2 - 2e_2$, and $L(n) = K(n+1) - e_1 K(n)$:
		\begin{align*}
			s_n &= K(n)(e_1^2 - 2e_2) + (K(n+1) - e_1 K(n))\,e_1 + M(n)\cdot 3 \\
			&= e_1 K(n+1) - 2e_2 K(n) + 3M(n).
		\end{align*}
		Substituting $M(n) = K(n+2) - e_1 K(n+1) + e_2 K(n)$ from Corollary~\ref{m_by_kl}
		(with the recurrence rearranged as $M(n) = L(n+1) + e_2 K(n) = K(n+2) - e_1 K(n+1) + e_2 K(n)$):
		\begin{align*}
			s_n &= e_1 K(n+1) - 2e_2 K(n) + 3\bigl(K(n+2) - e_1 K(n+1) + e_2 K(n)\bigr) \\
			&= 3K(n+2) - 2e_1 K(n+1) + e_2 K(n). \qedhere
		\end{align*}
	\end{proof}
