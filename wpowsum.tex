	\section{Setup and Conventions}
	\label{sec:setup}
	% ---------------------------------------------------------------------------
	
	Throughout, let $(x,y,z)\in\mathbb{R}^3$ be a fixed triplet with $xyz \neq 0$ and $x, y, z$ pairwise distinct.

	\subsection{Basic Definitions}
	
	\begin{definition}
		We define the elementary symmetric polynomials $e_1, e_2, e_3$ as
		\begin{align*}
			e_1 &= x + y + z = \cycsum{x}, \\
			e_2 &= xy + yz + zx = \cycsum{xy}, \\
			e_3 &= xyz. 
		\end{align*}
	\end{definition}
	
	\begin{definition}
		For any $n \in \mathbb{Z}$ and weights $(a, b, c) \in \mathbb{R}^3$, we define the weighted power sum $f_n(a, b, c)$ by
		\[ f_n(a, b, c) = ax^n + by^n + cz^n. \]
	\end{definition}
	
	\begin{definition}
		For any $n \in \mathbb{Z}$, let $s_n$ denote the unweighted power sum
		\[ s_n = f_n(1, 1, 1) = x^n + y^n + z^n = \cycsum{x^n}. \]
	\end{definition}
	
	\begin{definition}
		We define the weights $w_x, w_y, w_z$ as
		\begin{align*}
			w_x &= y - z, \\
			w_y &= z - x, \\
			w_z &= x - y.
		\end{align*}
	\end{definition}
	
	\begin{definition}
		We define $\Delta$ as the weighted power sum of degree 2:
		\[ \Delta = f_2(\wx, \wy, \wz) = \wx x^2 + \wy y^2 + \wz z^2 = \cycsum{\wx x^2}. \]
	\end{definition}
	
	\begin{lemma}[Factored Form of $\Delta$]
		\label{lem:delta_factored}
		$\Delta = -\wx\wy\wz.$
		In particular, $\Delta \neq 0$ since $x, y, z$ are pairwise distinct.
	\end{lemma}
	\begin{proof}
		Direct computation: $\cycsum{x^2 w_x} = x^2(y-z)+y^2(z-x)+z^2(x-y) = -(y-z)(z-x)(x-y) = -\wx\wy\wz$.
	\end{proof}

	\begin{definition}
		\label{def:char_poly}
		The \emph{characteristic polynomial} of $(x, y, z)$ is
		\[ P(t) = (t-x)(t-y)(t-z) = t^3 - e_1 t^2 + e_2 t - e_3. \]
	\end{definition}

	\subsection{The Sequences $K$, $L$, $M$}

	\begin{definition}
		\label{def:K}
		For each $n \in \mathbb{Z}$, we define the \emph{normalized power sum} $K(n)$ by
		\[ \Delta \cdot K(n) = f_n(\wx,\, \wy,\, \wz). \]
	\end{definition}

	\begin{definition}
		\label{def:L}
		For each $n \in \mathbb{Z}$, we define $L(n)$ by
		\[ \Delta \cdot L(n) = f_n(z^2 - y^2,\, x^2 - z^2,\, y^2 - x^2). \]
	\end{definition}

	\begin{definition}
		\label{def:M}
		For each $n \in \mathbb{Z}$, we define $M(n)$ by
		\[ \Delta \cdot M(n) = f_n(yz\cdot w_x,\, zx\cdot w_y,\, xy\cdot w_z). \]
	\end{definition}

	\begin{theorem}[Initial Values]
		\label{thm:initial_values}
		The initial values for $K$, $L$ and $M$ are:
		\begin{center}
			\begin{tabular}{c|ccc}
				$n$ & $K(n)$ & $L(n)$ & $M(n)$ \\
				\hline
				$0$ & $0$ & $0$ & $1$ \\
				$1$ & $0$ & $1$ & $0$ \\
				$2$ & $1$ & $0$ & $0$ \\
			\end{tabular}
		\end{center}
	\end{theorem}
	\begin{proof}
		Evaluate each entry via (\ref{def:K})--(\ref{def:M}) using $\cycsum{w_x} = 0$ and (\ref{lem:delta_factored}).
		\begin{align*}
			\Delta K(0) &= \cycsum{w_x} = 0, &
			\Delta L(0) &= \cycsum{(z^2-y^2)} = 0, &
			\Delta M(0) &= \cycsum{x^2 w_x} = \Delta, \\
			\Delta K(1) &= \cycsum{x\,w_x} = 0, &
			\Delta L(1) &= \cycsum{x(z^2-y^2)} = \Delta, &
			\Delta M(1) &= e_3\cycsum{w_x} = 0, \\
			\Delta K(2) &= \cycsum{x^2 w_x} = \Delta, &
			\Delta L(2) &= \cycsum{x^2(z^2-y^2)} = 0, &
			\Delta M(2) &= e_3\cycsum{x\,w_x} = 0.
		\end{align*}
		Dividing by $\Delta \neq 0$ gives the table.
	\end{proof}

	\subsection{Universal Recurrence}

	\begin{theorem}[Universal Recurrence]
		\label{thm:universal_recurrence}
		For any weights $(a, b, c)$ and any integer $n$,
		\[ f_{n+3}(a,b,c) = e_1\, f_{n+2}(a,b,c) - e_2\, f_{n+1}(a,b,c) + e_3\, f_n(a,b,c). \]
	\end{theorem}
	\begin{proof}
		Since $xyz \neq 0$, each root $t \in \{x,y,z\}$ is nonzero, so $t^n$ is defined for all $n \in \mathbb{Z}$. Each root satisfies $P(t) = 0$; multiply by $t^n$, then multiply by the respective weight and sum over $\{x,y,z\}$.
	\end{proof}

	\begin{corollary}[Universal Recurrence for $K$, $L$, $M$]
		\label{cor:recurrence_KLM}
		For any integer $n$:
		\begin{align*}
			K(n+3) &= e_1\,K(n+2) - e_2\,K(n+1) + e_3\,K(n), \\
			L(n+3) &= e_1\,L(n+2) - e_2\,L(n+1) + e_3\,L(n), \\
			M(n+3) &= e_1\,M(n+2) - e_2\,M(n+1) + e_3\,M(n).
		\end{align*}
	\end{corollary}
	\begin{proof}
		Each is defined as $\Delta^{-1} f_n(\cdot)$ with fixed weights (\ref{def:K})--(\ref{def:M}); apply (\ref{thm:universal_recurrence}) and divide by $\Delta \neq 0$.
	\end{proof}

	\section{Decomposition Theorem}
	\label{sec:decomposition}
	% ---------------------------------------------------------------------------
	
	\subsection{The Decomposition}
	\begin{theorem}[Decomposition]
		\label{thm:decomposition}
		For any weights $(a, b, c)$ and any integers $n, m$,
		\[ f_{n+m}(a,b,c) = K(n)\,f_{m+2}(a,b,c) + L(n)\,f_{m+1}(a,b,c) + M(n)\,f_m(a,b,c). \]
	\end{theorem}
	\begin{proof}
		Fix $m$ and $(a,b,c)$ and regard both sides as sequences in $n$. Both satisfy the same recurrence by (\ref{thm:universal_recurrence}) and (\ref{cor:recurrence_KLM}). Check equality at $n = 0, 1, 2$ using (\ref{thm:initial_values}); since a linear recurrence is uniquely determined by three consecutive values, the identity holds for all $n \in \mathbb{Z}$.
	\end{proof}
	
	\begin{corollary}[Decomposition at $m = 0$]
		\label{cor:decomp_m0}
		For any weights $(a, b, c)$ and any integer $n$,
		\[ f_n(a,b,c) = K(n)\,f_2(a,b,c) + L(n)\,f_1(a,b,c) + M(n)\,f_0(a,b,c). \]
	\end{corollary}
	\begin{proof}
		Set $m = 0$ in (\ref{thm:decomposition}).
	\end{proof}

	\subsection{Properties of $K$, $L$, $M$}

	\begin{corollary}[$s_n$ in terms of $K$]
		\label{cor:sn_K}
		For any integer $n$,
		\[ s_n = 3K(n+2) - 2e_1 K(n+1) + e_2 K(n). \]
	\end{corollary}
	\begin{proof}
		Apply (\ref{cor:decomp_m0}) with weights $(1,1,1)$: $s_n = K(n)\,s_2 + L(n)\,s_1 + M(n)\,s_0$. Substitute $s_0 = 3$, $s_1 = e_1$, $s_2 = e_1^2 - 2e_2$ and use (\ref{thm:initial_values}).
	\end{proof}

	\begin{corollary}[Small Values of $K$, $L$, $M$]
		\label{cor:small_values}
		The values of $K$, $L$, $M$ for $n = -1, \ldots, 4$ are:
		\begin{center}
			\begin{tabular}{c|ccc}
				$n$ & $K(n)$ & $L(n)$ & $M(n)$ \\
				\hline
				$-1$ & $e_3^{-1}$ & $-e_1 e_3^{-1}$ & $e_2 e_3^{-1}$ \\
				$0$  & $0$ & $0$ & $1$ \\
				$1$  & $0$ & $1$ & $0$ \\
				$2$  & $1$ & $0$ & $0$ \\
				$3$  & $e_1$ & $-e_2$ & $e_3$ \\
				$4$  & $e_1^2 - e_2$ & $e_3 - e_1 e_2$ & $e_1 e_3$ \\
			\end{tabular}
		\end{center}
	\end{corollary}
	\begin{proof}
		$K$: apply (\ref{cor:recurrence_KLM}) forward from (\ref{thm:initial_values}), and backward at $n=-1$ to get $K(-1) = e_3^{-1}$. $L$ and $M$: (\ref{def:L}) and (\ref{def:M}) evaluated directly, or equivalently via $L(n) = K(n+1) - e_1 K(n)$ and $M(n) = e_3 K(n-1)$.
	\end{proof}

	\begin{lemma}[Relations among $K$, $L$, $M$]
		\label{lem:KLM_relations}
		For all $n \in \mathbb{Z}$:
		\begin{align}
			L(n) &= K(n+1) - e_1\,K(n), \label{eq:L_by_K} \\
			M(n) &= e_3\,K(n-1), \label{eq:M_by_K} \\
			M(n) &= L(n+1) + e_2\,K(n). \label{eq:M_by_LK}
		\end{align}
	\end{lemma}
	\begin{proof}
		In each case both sides satisfy (\ref{cor:recurrence_KLM}), so it suffices to verify equality at $n = 0, 1, 2$. All values are read directly from (\ref{cor:small_values}).
	\end{proof}

	\begin{corollary}[Decomposition for $K$, $L$, $M$]
		\label{cor:decomp_K}
		For any integers $n, m$,
		\begin{align*}
			K(n+m) &= K(n)\,K(m+2) + L(n)\,K(m+1) + M(n)\,K(m), \\
			L(n+m) &= K(n)\,L(m+2) + L(n)\,L(m+1) + M(n)\,L(m), \\
			M(n+m) &= K(n)\,M(m+2) + L(n)\,M(m+1) + M(n)\,M(m).
		\end{align*}
	\end{corollary}
	\begin{proof}
		Apply (\ref{thm:decomposition}) with weights $(\wx, \wy, \wz)$ from (\ref{def:K}), $(z^2-y^2, x^2-z^2, y^2-x^2)$ from (\ref{def:L}), and $(yz\,w_x, zx\,w_y, xy\,w_z)$ from (\ref{def:M}) respectively, and divide throughout by $\Delta$.
	\end{proof}


	\subsection{Negative Index}

	\begin{lemma}[Antisymmetric weight evaluations]
		\label{lem:antisym_weights}
		For any integer $n$,
		\begin{align*}
			f_0(y^n-z^n,\, z^n-x^n,\, x^n-y^n) &= 0, \\
			f_1(y^n-z^n,\, z^n-x^n,\, x^n-y^n) &= -\Delta\, K(n), \\
			f_2(y^n-z^n,\, z^n-x^n,\, x^n-y^n) &= \Delta\, L(n).
		\end{align*}
	\end{lemma}
	\begin{proof}
		Line 1: $\cycsum{(y^n-z^n)} = 0$. Line 2: apply (\ref{cor:decomp_m0}) with weights $(\wx,\wy,\wz)$; using (\ref{def:K}) and $f_0(\wx,\wy,\wz) = 0$, $f_1(\wx,\wy,\wz) = 0$, $f_2(\wx,\wy,\wz) = \Delta$ from (\ref{thm:initial_values}) gives $-\Delta\,K(n)$. Line 3: the same with weights $(z^2-y^2, x^2-z^2, y^2-x^2)$ and (\ref{def:L}) gives $\Delta\,L(n)$.
	\end{proof}
	
	\begin{lemma}[Negative index expansion]
		\label{lem:neg_index_expand}
		For any integer $n$,
		\[ \Delta \cdot e_3^n \cdot K(-n) = f_{n+1}(y^n-z^n,\, z^n-x^n,\, x^n-y^n). \]
	\end{lemma}
	\begin{proof}
		By (\ref{def:K}), $\Delta \cdot e_3^n \cdot K(-n) = (xyz)^n f_{-n}(\wx,\wy,\wz) = \cycsum{(xyz)^n \wx x^{-n}} = \cycsum{y^nz^n(y-z)} = \cycsum{x^{n+1}(y^n-z^n)} = f_{n+1}(y^n-z^n, z^n-x^n, x^n-y^n)$.
	\end{proof}

	\begin{theorem}[Negative index]
		\label{thm:negative_index}
		For any integer $n$,
		\[ e_3^n \cdot K(-n) = K(n+1)\,L(n) - L(n+1)\,K(n). \]
	\end{theorem}
	\begin{proof}
		By (\ref{lem:neg_index_expand}), $\Delta \cdot e_3^n \cdot K(-n) = f_{n+1}(y^n-z^n,\, z^n-x^n,\, x^n-y^n)$.
		Apply (\ref{cor:decomp_m0}) with weights $(y^n-z^n, z^n-x^n, x^n-y^n)$; the $M(n+1)$ term vanishes by the first line of (\ref{lem:antisym_weights}), leaving
		\[ \Delta \cdot e_3^n \cdot K(-n) = K(n+1)\,f_2(y^n-z^n, \ldots) + L(n+1)\,f_1(y^n-z^n, \ldots). \]
		Substitute (\ref{lem:antisym_weights}) and divide by $\Delta$.
	\end{proof}
	
	\begin{corollary}[Negative Index via $K$ Only]
		\label{cor:negative_index_K}
		For any integer $n$,
		\[ e_3^n \cdot K(-n) = K(n+1)^2 - K(n+2)\,K(n). \]
	\end{corollary}
	\begin{proof}
		Expand $L$ and $L(n+1)$ using (\ref{eq:L_by_K}) from (\ref{lem:KLM_relations}) in (\ref{thm:negative_index}); the $e_1$ terms cancel.
	\end{proof}
	
	\subsection{The $d$-Sequences}

	\begin{definition}
		For any $n \in \mathbb{Z}$, we define
		\begin{align*}
			d_x(n) &= K(n+1) - x\,K(n), \\
			d_y(n) &= K(n+1) - y\,K(n), \\
			d_z(n) &= K(n+1) - z\,K(n).
		\end{align*}
	\end{definition}

	\begin{definition}
		For any $n \in \mathbb{Z}$, we define
		\[ D(n) = d_x(n) + d_y(n) + d_z(n). \]
	\end{definition}

	\begin{lemma}[Weighted Power Sum Representations of $d$-Sequences]
		\label{lem:D_weights}
		For any integer $n$:
		\begin{align*}
			\Delta\cdot d_x(n) &= f_n(0,\, -w_y w_z,\, w_y w_z), \\[4pt]
			\Delta\cdot d_y(n) &= f_n(w_z w_x,\, 0,\, -w_z w_x), \\[4pt]
			\Delta\cdot d_z(n) &= f_n(-w_x w_y,\, w_x w_y,\, 0).
		\end{align*}
	\end{lemma}
	\begin{proof}
		Both sides are weighted power sums with fixed weights, so by (\ref{thm:universal_recurrence}) it suffices to verify equality at $n = 0, 1, 2$; this follows by direct substitution using (\ref{thm:initial_values}) and (\ref{cor:small_values}). The formulas for $d_y$ and $d_z$ follow by cyclic symmetry.
	\end{proof}

	\begin{lemma}[Closed Form of $d$-Sequences]
		\label{lem:D_closed_form}
		For any integer $n$,
		\begin{align*}
			d_x(n) &= \frac{y^n - z^n}{y - z}, \\[4pt]
			d_y(n) &= \frac{z^n - x^n}{z - x}, \\[4pt]
			d_z(n) &= \frac{x^n - y^n}{x - y}.
		\end{align*}
	\end{lemma}
	\begin{proof}
		From (\ref{lem:D_weights}), $\Delta\cdot d_x(n) = f_n(0, -w_yw_z, w_yw_z) = -w_yw_z(y^n - z^n)$. Since $\Delta = -(y-z)\cdot w_yw_z$ from (\ref{lem:delta_factored}), dividing gives $d_x(n) = (y^n-z^n)/(y-z)$. The formulas for $d_y(n)$ and $d_z(n)$ follow by cyclic symmetry.
	\end{proof}

	\begin{corollary}[$D$-Sum Identity]
		\label{cor:D_sum}
		For any integer $n$,
		\[ D(n) = 3K(n+1) - e_1\,K(n). \]
	\end{corollary}
	\begin{proof}
		Add the three definitions directly.
	\end{proof}

	\begin{definition}
		\label{def:Gdelta}
		For any integer $n$, we define $\Gdelta{n}$ by
		\[ \Gdelta{n} = d_x(n)\,d_y(n)\,d_z(n). \]
	\end{definition}

	\begin{theorem}[$\Gamma$ via Characteristic Polynomial]
		\label{thm:gamma_char_poly}
		For any integer $n$ with $K(n) \neq 0$,
		\[
			\Gdelta{n} = K(n)^3 \cdot P\!\left(\frac{K(n+1)}{K(n)}\right).
		\]
	\end{theorem}
	\begin{proof}
		Substituting $d_x(n) = K(n+1) - x\,K(n)$ cyclically, $\Gdelta{n} = \cycprod{(K(n+1) - x\,K(n))} = K(n)^3\cycprod{(K(n+1)/K(n) - x)} = K(n)^3 \cdot P(K(n+1)/K(n))$.
	\end{proof}

	\begin{corollary}[Expanded $\Gamma$ Identity]
		\label{cor:gamma_expanded}
		For any integer $n$,
		\[
			\Gdelta{n} = K(n+1)^3 - e_1\,K(n+1)^2\,K(n) + e_2\,K(n+1)\,K(n)^2 - e_3\,K(n)^3.
		\]
	\end{corollary}
	\begin{proof}
		Expand $K(n)^3 \cdot P(K(n+1)/K(n))$ from (\ref{thm:gamma_char_poly}) using $P(t) = t^3 - e_1 t^2 + e_2 t - e_3$ from (\ref{def:char_poly}).
	\end{proof}

	% ---------------------------------------------------------------------------
	\section{The Power Lift}
	\label{sec:power_lift}
	% ---------------------------------------------------------------------------

	For any integer $k$, the triplet $(x^k, y^k, z^k)$ satisfies the same assumptions as $(x, y, z)$
	(nonzero product, pairwise distinct), so the entire framework of the preceding sections applies
	with $(x^k, y^k, z^k)$ in place of $(x, y, z)$. For any expression $Q$
	built from $(x, y, z)$, we write $\lift{Q}{k}$ for the corresponding expression in $(x^k, y^k, z^k)$.

	\subsection{Power Lift Properties}
	\begin{proposition}[Power lift is a ring homomorphism]
		\label{prop:plift_homomorphism}
		For any integer $k$ and any expressions $A$, $B$ built from $(x, y, z)$,
		\begin{align*}
			\lift{A + B}{k} &= \lift{A}{k} + \lift{B}{k}, \\
			\lift{A \cdot B}{k} &= \lift{A}{k} \cdot \lift{B}{k}.
		\end{align*}
	\end{proposition}
	\begin{proof}
		Both operations act pointwise on $x^k, y^k, z^k$.
	\end{proof}
	
	\begin{proposition}[Lift commutes with sequence evaluation]
		\label{prop:lift_commutes_eval}
		For any integers $k$ and $n$,
		\[ \lift{K(n)}{k} = \lift{K}{k}(n), \]
		and the same identity holds for $L(n)$ and $M(n)$.
	\end{proposition}
	\begin{proof}
		Write $\Delta \cdot K(n) = f_n(\wx, \wy, \wz)$ and distribute the lift using (\ref{prop:plift_homomorphism}); the result is $\lift{K}{k}(n)$ by definition.
	\end{proof}
	
	\begin{proposition}[Composition of lifts]
		\label{prop:lift_composition}
		For any integers $a$, $b$ and any expression $Q$ built from $(x, y, z)$,
		\[ \lift{\lift{Q}{a}}{b} = \lift{Q}{ab}. \]
	\end{proposition}
	\begin{proof}
		The inner lift sends $t \mapsto t^a$; the outer lift then sends $t^a \mapsto (t^a)^b = t^{ab}$, giving $\lift{Q}{ab}$.
	\end{proof}
	
	\subsection{Lifted $K$ and $\Delta$}

	\begin{theorem}[$\Gamma$ via Lifted $\Delta$]
		\label{thm:gamma_ratio}
		For any integer $n$,
		\[ \Delta \cdot \Gdelta{n} = \lift{\Delta}{n}. \]
	\end{theorem}
	\begin{proof}
		By (\ref{lem:D_closed_form}), $\Gdelta{n} = \cycprod{(y^n-z^n)/(y-z)}$. Multiplying by $\Delta = -\cycprod{(y-z)}$ from (\ref{lem:delta_factored}) cancels the denominators: $\Delta\cdot\Gdelta{n} = -\cycprod{(y^n-z^n)} = \lift{\Delta}{n}$.
	\end{proof}

	\begin{proposition}[Weighted Power Sum under Lift]
		\label{prop:lifted_K_def}
		For any integers $k$ and $n$,
		\[ \lift{\Delta}{n}\lift{K}{n}(k) = f_{kn}(\lift{\wx}{n},\, \lift{\wy}{n},\, \lift{\wz}{n}). \]
	\end{proposition}
	\begin{proof}
		By definition $\lift{\Delta}{n} \cdot \lift{K}{n}(k) = f_k(\lift{\wx}{n}, \lift{\wy}{n}, \lift{\wz}{n})$ in the lifted triplet. Since $f_k$ in the triplet $(x^n, y^n, z^n)$ equals $f_{kn}$ in the original variables, the result follows.
	\end{proof}
	
	\begin{lemma}[Lifted Weight Evaluations]
		\label{lem:lifted_weights_eval}
		For any integer $n$,
		\begin{align*}
			f_0(\lift{\wx}{n},\, \lift{\wy}{n},\, \lift{\wz}{n}) &= 0, \\
			f_1(\lift{\wx}{n},\, \lift{\wy}{n},\, \lift{\wz}{n}) &= -\Delta\, K(n), \\
			f_2(\lift{\wx}{n},\, \lift{\wy}{n},\, \lift{\wz}{n}) &= \Delta\, L(n).
		\end{align*}
	\end{lemma}
	\begin{proof}
		Since $\lift{\wx}{n} = y^n - z^n$, $\lift{\wy}{n} = z^n - x^n$, $\lift{\wz}{n} = x^n - y^n$, this is a direct restatement of (\ref{lem:antisym_weights}) in lift notation.
	\end{proof}
	
	\begin{theorem}[Lifted $K$ Connection]
		\label{thm:lifted_K_connection}
		For any integers $k$ and $n$,
		\[
		\Gdelta{n} \cdot \lift{K}{n}(k)
		= K(kn)\,L(n) - L(kn)\,K(n).
		\]
	\end{theorem}
	\begin{proof}
		Apply (\ref{prop:lifted_K_def}) to expand the left-hand side as $f_{kn}(\lift{\wx}{n}, \lift{\wy}{n}, \lift{\wz}{n})$, then decompose using (\ref{cor:decomp_m0}) and substitute (\ref{lem:lifted_weights_eval}); divide throughout by $\Delta$.
	\end{proof}
	
	\begin{corollary}[Lifted $K$ in Terms of $K$ Only]
		\label{cor:lifted_K_pure}
		For any integers $k$ and $n$,
		\[
		\Gdelta{n} \cdot \lift{K}{n}(k)
		= K(kn)\,K(n+1) - K(kn+1)\,K(n).
		\]
	\end{corollary}
	\begin{proof}
		Apply (\ref{thm:lifted_K_connection}) and expand $L(n) = K(n+1) - e_1 K(n)$ and $L(kn) = K(kn+1) - e_1 K(kn)$ using (\ref{eq:L_by_K}); the $e_1 K(kn) K(n)$ terms cancel.
	\end{proof}
	
	\begin{corollary}[$\Gamma$ via $K$]
		\label{cor:delta_ratio}
		For any integer $n$,
		\[ \Gdelta{n} = K(2n)\,K(n+1) - K(2n+1)\,K(n). \]
	\end{corollary}
	\begin{proof}
		Set $k = 2$ in (\ref{cor:lifted_K_pure}) and use $\lift{K}{n}(2) = 1$ from (\ref{cor:small_values}) applied in the lifted triplet via (\ref{prop:lift_commutes_eval}).
	\end{proof}
	
	\begin{corollary}[$s_n$ via lifted $\Delta$]
		\label{cor:sn_lifted}
		For any integer $n$,
		\[ \Gdelta{n} \cdot s_n = K(3n)\,K(n+1) - K(3n+1)\,K(n). \]
	\end{corollary}
	\begin{proof}
		Set $k = 3$ in (\ref{cor:lifted_K_pure}) and use $\lift{K}{n}(3) = \lift{s_1}{n} = s_n$ from (\ref{cor:sn_K}) lifted via (\ref{prop:lift_commutes_eval}).
	\end{proof}
	
	\subsection{$\Gamma$ Multiplicativity}
	\begin{theorem}[$\Gamma$ multiplicativity]
		\label{thm:gamma_mult}
		For any integers $n$ and $m$,
		\[ \Gdelta{nm} = \lift{\Gdelta{m}}{n} \cdot \Gdelta{n}. \]
	\end{theorem}
	\begin{proof}
		Expand each factor using (\ref{thm:gamma_ratio}) and (\ref{prop:lift_composition}): $\lift{\Gdelta{m}}{n} = \lift{\Delta}{nm}/\lift{\Delta}{n}$ and $\Gdelta{n} = \lift{\Delta}{n}/\Delta$; their product telescopes to $\lift{\Delta}{nm}/\Delta = \Gdelta{nm}$.
	\end{proof}
	
	\subsection{Further Lift Identities}

	\begin{lemma}[Negative lift of elementary symmetrics]
		\label{lem:neg_lift_esym}
		\begin{align*}
			e_3 \cdot \lift{e_2}{-1} &= e_1, \\
			e_3 \cdot \lift{e_1}{-1} &= e_2, \\
			e_3 \cdot \lift{e_3}{-1} &= 1.
		\end{align*}
	\end{lemma}
	\begin{proof}
		Substitute $(x^{-1}, y^{-1}, z^{-1})$ into each $e_k$ and multiply through by $e_3 = xyz$.
	\end{proof}
	
	\begin{theorem}[Lifted $L$ via $K$]
		\label{thm:lifted_L_pure}
		For any integers $k$ and $n$,
		\[ \Gdelta{n} \cdot \lift{L}{n}(k) = K(kn+1)\,K(2n) - K(kn)\,K(2n+1). \]
	\end{theorem}
	\begin{proof}
		By (\ref{def:L}) in the lifted triplet and (\ref{thm:gamma_ratio}), $\Gdelta{n}\cdot\lift{L}{n}(k) = \Delta^{-1} f_{kn}(z^{2n}-y^{2n}, x^{2n}-z^{2n}, y^{2n}-x^{2n})$. Apply (\ref{cor:decomp_m0}); the $f_0$ term vanishes by $\cycsum{(z^{2n}-y^{2n})}=0$, and (\ref{lem:antisym_weights}) at $2n$ with reversed sign gives $f_1 = \Delta K(2n)$ and $f_2 = -\Delta L(2n)$, so $\Gdelta{n}\cdot\lift{L}{n}(k) = K(2n)\,L(kn) - L(2n)\,K(kn)$. Expand both $L$-values via (\ref{eq:L_by_K}); the $e_1$ terms cancel.
	\end{proof}

	\subsection{General Decimation}

	\begin{theorem}[General Master Identity]
		\label{thm:general_master}
		For any integers $n$ and $j$:
		\[ K(3n+j) = s_n\,K(2n+j) - \lift{e_2}{n}\,K(n+j) + e_3^n\,K(j). \]
	\end{theorem}
	\begin{proof}
		The lifted characteristic polynomial $\lift{P}{n}(t)$ has roots $x^n, y^n, z^n$; each root satisfies
		\[ t^{3n} - \lift{e_1}{n}\,t^{2n} + \lift{e_2}{n}\,t^{n} - \lift{e_3}{n} = 0 \]
		by (\ref{def:char_poly}) and (\ref{prop:plift_homomorphism}). Since $\lift{e_1}{n} = s_n$ and $\lift{e_3}{n} = e_3^n$ (as $e_1 = s_1$), multiplying by $w_t\,t^j$ and summing over $\{x,y,z\}$ gives $\Delta\,K(3n+j) = s_n\,\Delta\,K(2n+j) - \lift{e_2}{n}\,\Delta\,K(n+j) + e_3^n\,\Delta\,K(j)$ by (\ref{def:K}). Divide by $\Delta \neq 0$.
	\end{proof}

	\begin{theorem}[General Decimation Identity]
		\label{thm:general_decimation}
		For any integers $n$, $m$, and $j$:
		\[
			K(nm+j)
			= \lift{K}{n}(m)\,K(2n+j)
			  + \lift{L}{n}(m)\,K(n+j)
			  + \lift{M}{n}(m)\,K(j).
		\]
	\end{theorem}
	\begin{proof}
		Fix $n$ and $j$; regard both sides as sequences in $m$. Both satisfy the lifted recurrence with coefficients $\lift{e_1}{n}, \lift{e_2}{n}, \lift{e_3}{n}$: the right-hand side by (\ref{cor:recurrence_KLM}) in the lifted triplet, the left-hand side by (\ref{thm:general_master}) with $j$ replaced by $n(m-2)+j$. Equality at $m = 0, 1, 2$ follows from (\ref{thm:initial_values}) via (\ref{prop:lift_commutes_eval}); uniqueness of the recurrence extends it to all $m \in \mathbb{Z}$.
	\end{proof}

	\begin{remark}
		Setting $m = 3$ and substituting $\lift{K}{n}(3) = s_n$, $\lift{L}{n}(3) = -\lift{e_2}{n}$, $\lift{M}{n}(3) = e_3^n$ from (\ref{cor:small_values}) recovers (\ref{thm:general_master}).
	\end{remark}

	\section{The Zero Sum Assumption}
	\label{sec:sn_zero}
	% ---------------------------------------------------------------------------
	
	Throughout this section, $n$ denotes a fixed integer satisfying $s_n = 0$.
	
	\subsection{Lifted Symmetrics at Zero Sum}
	\begin{corollary}[Lifted Symmetrics under Zero Sum]
		\label{cor:lifted_e1e2}
		\begin{align*}
			\lift{e_1}{n} &= 0\\
			\lift{e_2}{n} &= \cycsum{x^ny^n}\\
			\lift{e_3}{n} &= e_3^n
		\end{align*}
	\end{corollary}
	\begin{proof}
		Both follow directly from $s_n = 0$ and the definition of the power lift.
	\end{proof}
	
	\begin{corollary}[Lifted Power Sums under Zero Sum]
		\label{cor:lifted_s_values}
		\begin{align*}
			\lift{s_1}{n} &= 0\\
			\lift{s_2}{n} &= -2\lift{e_2}{n}\\
			\lift{s_3}{n} &= 3e_3^n\\
			\lift{s_4}{n} &= 2\lift{e_2}{n}^2
		\end{align*}
	\end{corollary}
	\begin{proof}
		Follows directly from lifting and using $s_n = 0$.
	\end{proof}
	
	\begin{corollary}[Lifted $K$, $L$, $M$ Values under Zero Sum]
		\label{cor:lifted_k_values}
		The values of $\lift{K}{n}$, $\lift{L}{n}$, $\lift{M}{n}$ for $k = 3, \ldots, 6$ are:
		\begin{center}
			\begin{tabular}{c|ccc}
				$k$ & $\lift{K}{n}(k)$ & $\lift{L}{n}(k)$ & $\lift{M}{n}(k)$ \\
				\hline
				$3$ & $0$ & $-\lift{e_2}{n}$ & $e_3^n$ \\
				$4$ & $-\lift{e_2}{n}$ & $e_3^n$ & $0$ \\
				$5$ & $e_3^n$ & $\lift{e_2}{n}^2$ & $-\lift{e_2}{n}\,e_3^n$ \\
				$6$ & $\lift{e_2}{n}^2$ & $-\lift{e_2}{n}\,e_3^n$ & $e_3^{2n}$ \\
			\end{tabular}
		\end{center}
	\end{corollary}
	\begin{proof}
		The $K$ values follow by applying (\ref{cor:sn_K}) in the lifted triplet using $\lift{e_1}{n} = 0$ and the values of $\lift{s_k}{n}$ from (\ref{cor:lifted_s_values}). The $L$ and $M$ values then follow from (\ref{eq:L_by_K}) and (\ref{eq:M_by_K}) lifted to $n$, using $\lift{e_1}{n} = 0$ and $\lift{e_3}{n} = e_3^n$ from (\ref{cor:lifted_e1e2}).
	\end{proof}
	
	\begin{corollary}[Negative Lift under Zero Sum]
		\label{cor:neg_lift_esym_zero_sum}
		\begin{align*}
			e_3^n \cdot \lift{e_1}{-n} &= \lift{e_2}{n}, \\
			\lift{e_2}{-n} &= 0, \\
			e_3^n \cdot \lift{e_3}{-n} &= 1.
		\end{align*}
	\end{corollary}
	\begin{proof}
		Lift each identity in (\ref{lem:neg_lift_esym}) to $n$ using (\ref{prop:plift_homomorphism}). Then use $s_n = 0$ and $\lift{e_3}{n} = e_3^n$
		from (\ref{cor:lifted_e1e2}).
	\end{proof}

	\subsection{The $d$-Sequences under Zero Sum}

	\begin{corollary}[Lifted $D$-Sum under Zero Sum]
		\label{cor:D_sum_zero}
		For any integer $k$,
		\[ \lift{D}{n}(k) = 3\lift{K}{n}(k+1). \]
	\end{corollary}
	\begin{proof}
		Apply (\ref{cor:D_sum}) in the lifted triplet $(x^n, y^n, z^n)$ and use $\lift{e_1}{n} = 0$ from (\ref{cor:lifted_e1e2}).
	\end{proof}

	\subsection{The Master Identity}

	\begin{corollary}[Master Identity under Zero Sum]
		\label{cor:master_zero_sum}
		For any integer $j$:
		\[ K(3n+j) = -\lift{e_2}{n}\cdot K(n+j) + e_3^n\cdot K(j). \]
	\end{corollary}
	\begin{proof}
		Set $s_n = 0$ in Theorem~\ref{thm:general_master}.
	\end{proof}

	\begin{corollary}[Explicit Tripling of $K$]
		\label{cor:K_tripling}
		\begin{align*}
			K(3n) &= -\lift{e_2}{n}\cdot K(n), \\
			K(3n+1) &= -\lift{e_2}{n}\cdot K(n+1).
		\end{align*}
	\end{corollary}
	\begin{proof}
		Set $j = 0$ and $j = 1$ in (\ref{cor:master_zero_sum}), using $K(0) = K(1) = 0$ from (\ref{thm:initial_values}).
	\end{proof}

	\begin{corollary}[Negative Tripling of $K$]
		\label{cor:K_neg_tripling}
		For any integer $n$:
		\begin{align*}
			e_3^n \cdot K(-3n)   &= \lift{e_2}{n} \cdot K(-2n), \\
			e_3^n \cdot K(-3n+1) &= \lift{e_2}{n} \cdot K(-2n+1).
		\end{align*}
	\end{corollary}
	\begin{proof}
		Set $j = -3n$ and $j = -3n+1$ in (\ref{cor:master_zero_sum}), using $K(0) = K(1) = 0$ from (\ref{thm:initial_values}).
	\end{proof}

	\begin{corollary}[Doubling of $K$ via Negative Index]
		\label{cor:K_doubling_neg}
		For any integer $n$:
		\begin{align*}
			K(2n)   &= e_3^n\cdot K(-n), \\
			K(2n+1) &= e_3^n\cdot K(-n+1).
		\end{align*}
	\end{corollary}
	\begin{proof}
		Set $j = -n$ and $j = -n+1$ in (\ref{cor:master_zero_sum}), using $K(0) = K(1) = 0$ from (\ref{thm:initial_values}).
	\end{proof}

	\begin{corollary}[Doubling of $K$ via Near Values]
		\label{cor:K_doubling_near}
		For any integer $n$:
		\begin{align*}
			K(2n)   &= K(n+1)^2 - K(n+2)\,K(n), \\
			K(2n+1) &= e_1\,K(n+1)^2 - K(n+1)\,K(n+2) - e_2\,K(n)\,K(n+1) + e_3\,K(n)^2.
		\end{align*}
	\end{corollary}
	\begin{proof}
		Apply (\ref{cor:negative_index_K}) at $n$ and at $n-1$ respectively, then substitute into (\ref{cor:K_doubling_neg}). For $K(2n+1)$, eliminate $e_3\,K(n-1)$ using the recurrence (\ref{cor:recurrence_KLM}): $e_3\,K(n-1) = K(n+2) - e_1\,K(n+1) + e_2\,K(n)$.
	\end{proof}

	\begin{corollary}[Recurrence under Zero Sum]
		\label{cor:sn_zero_K}
		\[ 3K(n+2) = 2e_1\,K(n+1) - e_2\,K(n). \]
	\end{corollary}
	\begin{proof}
		Set $s_n = 0$ in (\ref{cor:sn_K}) and rearrange.
	\end{proof}

	\begin{corollary}[Doubling of $K$ via Two Values]
		\label{cor:K_doubling_two}
		For any integer $n$:
		\begin{align*}
			3K(2n)   &= 3K(n+1)^2 - 2e_1 K(n)K(n+1) + e_2 K(n)^2, \\
			3K(2n+1) &= e_1 K(n+1)^2 - 2e_2 K(n)K(n+1) + 3e_3 K(n)^2.
		\end{align*}
	\end{corollary}
	\begin{proof}
		Multiply (\ref{cor:K_doubling_near}) by $3$ and substitute $3K(n+2) = 2e_1 K(n+1) - e_2 K(n)$ from (\ref{cor:sn_zero_K}).
	\end{proof}
