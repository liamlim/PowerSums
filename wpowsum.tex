	% ---------------------------------------------------------------------------
	\section{Setup and Conventions}
	\label{sec:setup}
	% ---------------------------------------------------------------------------
	
	Throughout, let $(x,y,z)\in\mathbb{R}^3$ be a fixed triplet with $xyz \neq 0$ and $x, y, z$ pairwise distinct.
	
	\begin{definition}
		We define the elementary symmetric polynomials $e_1, e_2, e_3$ as
		\begin{align*}
			e_1 &= x + y + z = \cycsum{x}, \\
			e_2 &= xy + yz + zx = \cycsum{xy}, \\
			e_3 &= xyz. 
		\end{align*}
	\end{definition}
	
	\begin{definition}
		For any $n \in \mathbb{Z}$ and weights $(a, b, c) \in \mathbb{R}^3$, we define the weighted power sum $f_n(a, b, c)$ by
		\[ f_n(a, b, c) = ax^n + by^n + cz^n. \]
	\end{definition}
	
	\begin{definition}
		For any $n \in \mathbb{Z}$, let $s_n$ denote the unweighted power sum
		\[ s_n = f_n(1, 1, 1) = x^n + y^n + z^n = \cycsum{x^n}. \]
	\end{definition}
	
	\begin{definition}
		We define the weights $w_x, w_y, w_z$ as
		\begin{align*}
			w_x &= y - z, \\
			w_y &= z - x, \\
			w_z &= x - y.
		\end{align*}
	\end{definition}
	
	\begin{theorem}[Vanishing Sums]
		\label{thm:vanishing}
		$f_0(w_x, w_y, w_z) = 0$ and $f_1(w_x, w_y, w_z) = 0$.
	\end{theorem}
	\begin{proof}
		Substitute $w_x, w_y, w_z$ and collect terms: $f_0 = (y-z)+(z-x)+(x-y) = 0$, and $f_1 = x(y-z) + y(z-x) + z(x-y) = 0$.
	\end{proof}
	
	\begin{definition}
		We define $\Delta$ as the weighted power sum of degree 2:
		\[ \Delta = f_2(w_x, w_y, w_z) = w_x x^2 + w_y y^2 + w_z z^2 = \cycsum{w_x x^2}. \]
	\end{definition}
	
	\begin{lemma}
		\label{lem:delta_factored}
		$\Delta = -w_x w_y w_z.$
		In particular, $\Delta \neq 0$ since $x, y, z$ are pairwise distinct.
	\end{lemma}
	\begin{proof}
		Expand $-(y-z)(z-x)(x-y)$ directly and collect terms to recover $w_x x^2 + w_y y^2 + w_z z^2 = \Delta$.
	\end{proof}
	
	\begin{definition}
		For each $n \in \mathbb{Z}$, we define the normalized power sum $K(n)$ by
		\[ K(n) = \frac{f_n(w_x, w_y, w_z)}{\Delta}. \]
	\end{definition}
	
	% ---------------------------------------------------------------------------
	\section{Universal Recurrence}
	\label{sec:recurrence}
	% ---------------------------------------------------------------------------
	
	\begin{definition}
		The \emph{characteristic polynomial} of $(x, y, z)$ is
		\[ P(t) = (t-x)(t-y)(t-z) = t^3 - e_1 t^2 + e_2 t - e_3. \]
	\end{definition}
	
	\begin{theorem}[Universal Recurrence]
		\label{thm:universal_recurrence}
		For any weights $(a, b, c)$ and any integer $n$,
		\[ f_{n+3} = e_1 f_{n+2} - e_2 f_{n+1} + e_3 f_n. \]
	\end{theorem}
	
	\begin{proof}
		Since $x, y, z$ are roots of $P(t) = 0$ and each is nonzero, multiply $P(t) = 0$ by $t^n$ to get the recurrence for each root individually; then multiply by the respective weights $a, b, c$ and sum over $\{x, y, z\}$ to obtain the result.
	\end{proof}
	
	\begin{corollary}[Recurrence of $K$]
		\label{cor:recurrence_K}
		The normalized sequence $K(n)$ satisfies the same recurrence:
		\[ K(n+3) = e_1 K(n+2) - e_2 K(n+1) + e_3 K(n). \]
	\end{corollary}
	
	\begin{proof}
		$K(n) = f_n(w_x, w_y, w_z) / \Delta$ and $\Delta \neq 0$ is constant, so dividing (\ref{thm:universal_recurrence}) throughout by $\Delta$ gives the same recurrence for $K(n)$.
	\end{proof}
	
	\subsection{Companion Sequences}
	\begin{definition}
		We define the companion sequences $L(n)$ and $M(n)$ for all $n \in \mathbb{Z}$ by:
		\begin{align*}
			L(n) &= K(n+1) - e_1 K(n), \\
			M(n) &= e_3 K(n-1).
		\end{align*}
	\end{definition}
	
	\begin{corollary}[Recurrence of $L$ and $M$]
		\label{cor:recurrence_LM}
		Both companion sequences satisfy the same recurrence:
		\[ L(n+3) = e_1 L(n+2) - e_2 L(n+1) + e_3 L(n), \]
		\[ M(n+3) = e_1 M(n+2) - e_2 M(n+1) + e_3 M(n). \]
	\end{corollary}
	\begin{proof}
		For $L$: substitute the definition $L(n) = K(n+1) - e_1 K(n)$ into the right-hand side and apply (\ref{cor:recurrence_K}) to reduce to the left-hand side. For $M$: substitute $M(n) = e_3 K(n-1)$, factor out $e_3$, and apply (\ref{cor:recurrence_K}) with index shifted by $-1$.
	\end{proof}
	
	\begin{corollary}[Alternative expression for $M$]
		\label{cor:m_by_kl}
		For all $n \in \mathbb{Z}$,
		\[ M(n) = L(n+1) + e_2 K(n). \]
	\end{corollary}
	\begin{proof}
		Substitute $L(n+1) = K(n+2) - e_1 K(n+1)$ and $M(n) = e_3 K(n-1)$, then apply (\ref{cor:recurrence_K}) to replace $e_3 K(n-1)$ with $K(n+2) - e_1 K(n+1) - e_2 K(n)$.
	\end{proof}
	
	% ---------------------------------------------------------------------------
	\subsection{Initial Values for K, L, M}
	\label{sec:initial}
	% ---------------------------------------------------------------------------
	
	We now compute $K$, $L$, $M$ at $n = 0, 1, 2$.
	
	\begin{theorem}[Initial Values]
		The initial values for $K$, $L$ and $M$ are:
		\label{thm:initial_values}
		\begin{center}
			\begin{tabular}{c|ccc}
				$n$ & $K(n)$ & $L(n)$ & $M(n)$ \\
				\hline
				$0$ & $0$ & $0$ & $1$ \\
				$1$ & $0$ & $1$ & $0$ \\
				$2$ & $1$ & $0$ & $0$ \\
			\end{tabular}
		\end{center}
	\end{theorem}
	\begin{proof}
		By (\ref{thm:vanishing}), the numerators $f_0$ and $f_1$ vanish, so $K(0) = K(1) = 0$; and $f_2(w_x,w_y,w_z) = \Delta$ by definition, so $K(2) = 1$. The values of $L$ and $M$ follow by substituting into their definitions.
	\end{proof}
	
	% ---------------------------------------------------------------------------
	\section{Decomposition Theorem}
	\label{sec:decomposition}
	% ---------------------------------------------------------------------------
	
	\begin{theorem}[Decomposition]
		\label{thm:decomposition}
		For any weights $(a, b, c)$ and any integers $n, m$,
		\[ f_{n+m}(a,b,c) = K(n)\,f_{m+2}(a,b,c) + L(n)\,f_{m+1}(a,b,c) + M(n)\,f_m(a,b,c). \]
	\end{theorem}
	
	\begin{proof}
		Fix $m$ and $(a,b,c)$ and regard both sides as sequences in $n$. Both satisfy the same recurrence by (\ref{thm:universal_recurrence}), (\ref{cor:recurrence_K}), and (\ref{cor:recurrence_LM}). Check equality at $n = 0, 1, 2$ using (\ref{thm:initial_values}); since a linear recurrence is uniquely determined by three consecutive values, the identity holds for all $n \in \mathbb{Z}$.
	\end{proof}
	
	\begin{corollary}[Decomposition at $m = 0$]
		\label{cor:decomp_m0}
		For any weights $(a, b, c)$ and any integer $n$,
		\[ f_n(a,b,c) = K(n)\,f_2(a,b,c) + L(n)\,f_1(a,b,c) + M(n)\,f_0(a,b,c). \]
	\end{corollary}
	\begin{proof}
		Set $m = 0$ in (\ref{thm:decomposition}).
	\end{proof}
	
	\begin{corollary}[$s_n$ in terms of $K$]
		\label{cor:sn_K}
		For any integer $n$,
		\[ s_n = 3K(n+2) - 2e_1 K(n+1) + e_2 K(n). \]
	\end{corollary}
	\begin{proof}
		Apply (\ref{cor:decomp_m0}) with weights $(1,1,1)$ and simplify.
	\end{proof}
	
	\begin{corollary}[Decomposition for $K$]
		\label{cor:decomp_K}
		For any integers $n, m$,
		\[ K(n+m) = K(n)\,K(m+2) + L(n)\,K(m+1) + M(n)\,K(m). \]
	\end{corollary}
	\begin{proof}
		Apply (\ref{thm:decomposition}) with weights $(a,b,c) = (w_x, w_y, w_z)$ and divide throughout by $\Delta$.
	\end{proof}
	
	\begin{lemma}[Antisymmetric weight evaluations]
		\label{lem:antisym_weights}
		For any integer $n$,
		\begin{align*}
			f_0(y^n-z^n,\, z^n-x^n,\, x^n-y^n) &= 0, \\
			f_1(y^n-z^n,\, z^n-x^n,\, x^n-y^n) &= -\Delta\, K(n), \\
			f_2(y^n-z^n,\, z^n-x^n,\, x^n-y^n) &= \Delta\, L(n).
		\end{align*}
	\end{lemma}
	\begin{proof}
		The first line is (\ref{thm:vanishing}). For the second, expand directly and apply (\ref{cor:decomp_m0}) with weights $(w_x, w_y, w_z)$ to recognise the result as $-f_n(w_x, w_y, w_z) = -\Delta\,K(n)$. For the third, expand to get $f_n(z^2-y^2, x^2-z^2, y^2-x^2)$, then apply (\ref{cor:decomp_m0}) with weights $(z^2-y^2, x^2-z^2, y^2-x^2)$; one computes $f_1 = \Delta$ and $f_2 = 0$ for these weights by (\ref{thm:vanishing}) and direct substitution, so the decomposition reduces to $L(n)\cdot\Delta$.
	\end{proof}
	
	\begin{theorem}[Negative index]
		\label{thm:negative_index}
		For any integer $n$,
		\[ e_3^n \cdot K(-n) = K(n+1)\,L(n) - L(n+1)\,K(n). \]
	\end{theorem}
	\begin{proof}
		One verifies directly that $\Delta \cdot e_3^n \cdot K(-n) = f_{n+1}(y^n-z^n,\, z^n-x^n,\, x^n-y^n)$.
		Apply (\ref{cor:decomp_m0}) to the right-hand side with weights $(y^n-z^n, z^n-x^n, x^n-y^n)$; the $M(n+1)$ term vanishes by the first line of (\ref{lem:antisym_weights}), leaving
		\[ \Delta \cdot e_3^n \cdot K(-n) = K(n+1)\,f_2(y^n-z^n, \ldots) + L(n+1)\,f_1(y^n-z^n, \ldots). \]
		Substitute (\ref{lem:antisym_weights}) and divide by $\Delta$.
	\end{proof}
	
	\begin{corollary}
		\label{cor:negative_index_K}
		For any integer $n$,
		\[ e_3^n \cdot K(-n) = K(n+1)^2 - K(n+2)\,K(n). \]
	\end{corollary}
	\begin{proof}
		Expand $L(n) = K(n+1) - e_1 K(n)$ and $L(n+1) = K(n+2) - e_1 K(n+1)$ in (\ref{thm:negative_index}); the $e_1$ terms cancel.
	\end{proof}
	
	% ---------------------------------------------------------------------------
	\section{The Lifted Space}
	\label{sec:lifted}
	% ---------------------------------------------------------------------------
	
	For any integer $k$, the triplet $(x^k, y^k, z^k)$ satisfies the same assumptions as $(x, y, z)$
	(nonzero product, pairwise distinct), so the entire framework of the preceding sections applies
	with $(x^k, y^k, z^k)$ in place of $(x, y, z)$. For any expression $Q$
	built from $(x, y, z)$, we write $\lift{Q}{k}$ for the corresponding expression in $(x^k, y^k, z^k)$.
	
	\begin{definition}
		For any integer $k$, we define
		\[ \Gdelta{k} = \frac{\lift{\Delta}{k}}{\Delta}. \]
	\end{definition}
	
	\begin{proposition}[Lift is a ring homomorphism]
		\label{prop:lift_homomorphism}
		For any integer $k$ and any expressions $A$, $B$ built from $(x, y, z)$,
		\begin{align*}
			\lift{A + B}{k} &= \lift{A}{k} + \lift{B}{k}, \\
			\lift{A \cdot B}{k} &= \lift{A}{k} \cdot \lift{B}{k}.
		\end{align*}
	\end{proposition}
	\begin{proof}
		Both operations act pointwise on $x^k, y^k, z^k$.
	\end{proof}
	
	\begin{proposition}[Lift commutes with sequence evaluation]
		\label{prop:lift_commutes_eval}
		For any integers $k$ and $n$,
		\[ \lift{K(n)}{k} = \lift{K}{k}(n), \]
		and the same identity holds for $L(n)$ and $M(n)$.
	\end{proposition}
	\begin{proof}
		Write $K(n) = f_n(w_x, w_y, w_z)/\Delta$ and distribute the lift using (\ref{prop:lift_homomorphism}); the result is $\lift{K}{k}(n)$ by definition.
	\end{proof}
	
	\begin{proposition}[Composition of lifts]
		\label{prop:lift_composition}
		For any integers $a$, $b$ and any expression $Q$ built from $(x, y, z)$,
		\[ \lift{\lift{Q}{a}}{b} = \lift{Q}{ab}. \]
	\end{proposition}
	\begin{proof}
		The inner lift sends $t \mapsto t^a$; the outer lift then sends $t^a \mapsto (t^a)^b = t^{ab}$, giving $\lift{Q}{ab}$.
	\end{proof}
	
	\begin{proposition}
		\label{prop:lifted_K_def}
		For any integers $k$ and $n$,
		\[ \lift{\Delta}{n}\lift{K}{n}(k) = f_{kn}(\lift{w_x}{n},\, \lift{w_y}{n},\, \lift{w_z}{n}). \]
	\end{proposition}
	\begin{proof}
		By definition $\lift{K}{n}(k) = f_k(\lift{w_x}{n}, \lift{w_y}{n}, \lift{w_z}{n}) / \lift{\Delta}{n}$. Multiply by $\lift{\Delta}{n}$; since $f_k$ in the triplet $(x^n, y^n, z^n)$ equals $f_{kn}$ in the original variables, the result follows.
	\end{proof}
	
	\begin{lemma}
		\label{lem:lifted_weights_eval}
		For any integer $n$,
		\begin{align*}
			f_0(\lift{w_x}{n},\, \lift{w_y}{n},\, \lift{w_z}{n}) &= 0, \\
			f_1(\lift{w_x}{n},\, \lift{w_y}{n},\, \lift{w_z}{n}) &= -\Delta\, K(n), \\
			f_2(\lift{w_x}{n},\, \lift{w_y}{n},\, \lift{w_z}{n}) &= \Delta\, L(n).
		\end{align*}
	\end{lemma}
	\begin{proof}
		Since $\lift{w_x}{n} = y^n - z^n$, $\lift{w_y}{n} = z^n - x^n$, $\lift{w_z}{n} = x^n - y^n$, this is a direct restatement of (\ref{lem:antisym_weights}) in lift notation.
	\end{proof}
	
	\begin{theorem}
		\label{thm:lifted_K_connection}
		For any integers $k$ and $n$,
		\[
		\Gdelta{n} \cdot \lift{K}{n}(k)
		= K(kn)\,L(n) - L(kn)\,K(n).
		\]
	\end{theorem}
	
	\begin{proof}
		Apply (\ref{prop:lifted_K_def}) to expand the left-hand side as $f_{kn}(\lift{w_x}{n}, \lift{w_y}{n}, \lift{w_z}{n})$, then decompose using (\ref{cor:decomp_m0}) and substitute (\ref{lem:lifted_weights_eval}); divide throughout by $\Delta$.
	\end{proof}
	
	\begin{corollary}[Lifted $K$]
		\label{cor:lifted_K_pure}
		For any integers $k$ and $n$,
		\[
		\Gdelta{n} \cdot \lift{K}{n}(k)
		= K(kn)\,K(n+1) - K(kn+1)\,K(n).
		\]
	\end{corollary}
	\begin{proof}
		Apply (\ref{thm:lifted_K_connection}) and expand $L(n) = K(n+1) - e_1 K(n)$ and $L(kn) = K(kn+1) - e_1 K(kn)$; the $e_1 K(kn) K(n)$ terms cancel.
	\end{proof}
	
	\begin{corollary}[$\Delta$ ratio]
		\label{cor:delta_ratio}
		For any integer $n$,
		\[ \Gdelta{n} = K(2n)\,K(n+1) - K(2n+1)\,K(n). \]
	\end{corollary}
	\begin{proof}
		Set $k = 2$ in (\ref{cor:lifted_K_pure}) and use $\lift{K}{n}(2) = 1$ from (\ref{thm:initial_values}) applied in the lifted triplet via (\ref{prop:lift_commutes_eval}).
	\end{proof}
	
	\begin{corollary}[$s_n$ via lifted $\Delta$]
		\label{cor:sn_lifted}
		For any integer $n$,
		\[ \Gdelta{n} \cdot s_n = K(3n)\,K(n+1) - K(3n+1)\,K(n). \]
	\end{corollary}
	\begin{proof}
		Set $k = 3$ in (\ref{cor:lifted_K_pure}) and use (\ref{cor:sn_K}) in the lifted triplet via (\ref{prop:lift_commutes_eval}) to express $\lift{\Delta}{n} \cdot \lift{K}{n}(3) = \Gdelta{n} \cdot \Delta \cdot s_n$; dividing by $\Delta$ gives the result.
	\end{proof}
	
	\begin{theorem}[$\Gamma$ multiplicativity]
		\label{thm:gamma_mult}
		For any integers $n$ and $m$,
		\[ \Gdelta{nm} = \lift{\Gdelta{m}}{n} \cdot \Gdelta{n}. \]
	\end{theorem}
	\begin{proof}
		Expand each factor using the definition $\Gdelta{k} = \lift{\Delta}{k}/\Delta$ and (\ref{prop:lift_composition}): $\lift{\Gdelta{m}}{n} = \lift{\Delta}{nm}/\lift{\Delta}{n}$ and $\Gdelta{n} = \lift{\Delta}{n}/\Delta$; their product telescopes to $\lift{\Delta}{nm}/\Delta = \Gdelta{nm}$.
	\end{proof}
	
	\begin{lemma}
		\label{lem:gdelta3_two_variables}
		\[ \Gdelta{3} = (x^2+xy+y^2)(y^2+yz+z^2)(z^2+zx+x^2). \]
	\end{lemma}
	\begin{proof}
		It follows directly from the definition of $\Gdelta{}$ and the definition of $\Delta$.
	\end{proof}
	
	% ---------------------------------------------------------------------------
	\section{The Zero Sum Assumption}
	\label{sec:sn_zero}
	% ---------------------------------------------------------------------------
	
	Throughout this section we assume $s_n = 0$.
	
	\begin{corollary}
		\label{cor:lifted_e1e2}
		\begin{align*}
			\lift{e_1}{n} &= 0\\
			\lift{e_2}{n} &= \cycsum{x^ny^n}\\
			\lift{e_3}{n} &= e_3^n
		\end{align*}
	\end{corollary}
	\begin{proof}
		Both follow directly from $s_n = 0$ and the definition of the lift.
	\end{proof}
	
	\begin{corollary}
		\label{cor:lifted_s_values}
		\begin{align*}
			\lift{s_1}{n} &= 0\\
			\lift{s_2}{n} &= -2\lift{e_2}{n}\\
			\lift{s_3}{n} &= 3e_3^n\\
			\lift{s_4}{n} &= 2\lift{e_2}{n}^2
		\end{align*}
	\end{corollary}
	\begin{proof}
		Follows directly from lifting and using $s_n = 0$.
	\end{proof}
	
	\begin{corollary}
		\label{cor:lifted_k_values}
		\begin{align*}
			\lift{K(3)}{n} &= 0\\
			\lift{K(4)}{n} &= -\lift{e_2}{n}\\
			\lift{K(5)}{n} &= e_3^n\\
			\lift{K(6)}{n} &= \lift{e_2}{n}^2\\
		\end{align*}
	\end{corollary}
	\begin{proof}
		Apply (\ref{cor:sn_K}) in the lifted triplet $(x^n, y^n, z^n)$ using $\lift{e_1}{n} = 0$ and the values of $\lift{s_k}{n}$ from the preceding corollary.
	\end{proof}
	
	\begin{theorem}
		\label{thm:zero_sum_3n}
		\[ K(3n)\cdot K(n+1) = K(3n+1)\cdot K(n). \]
	\end{theorem}
	
	\begin{proof}
		Immediate from (\ref{cor:sn_lifted}), since $\Gdelta{n} \cdot s_n = 0$.
	\end{proof}
	
	\begin{definition}
		\label{def:A}
		For any integer $n$ with $K(n) \neq 0$, we define $A$ by
		\[ A \cdot K(n) = K(3n). \]
	\end{definition}
	
	\begin{corollary}
		\label{cor:A_shift}
		For any integer $n$ with $K(n) \neq 0$,
		\[ A \cdot K(n+1) = K(3n+1). \]
	\end{corollary}
	
	\begin{proof}
		Follows from (\ref{thm:zero_sum_3n}) and (\ref{def:A}).
	\end{proof}
	
	\begin{lemma}
		\label{lem:k3n}
		For any integer $m$,
		\[ \lift{e_2}{n}^2\cdot \lift{K}{3n}(m) = \lift{K}{n}(3m). \]
	\end{lemma}
	\begin{proof}
		Set $n := 3$, $k := m$ in (\ref{cor:lifted_K_pure}), lift to $n$ using (\ref{prop:lift_composition}), then substitute (\ref{cor:lifted_e1e2}) and (\ref{cor:lifted_k_values}).
	\end{proof}
	
	\begin{theorem}
		\label{thm:gamma_3n}
		\[ \Gdelta{n}\cdot \lift{e_2}{n}^2\cdot A = \Gdelta{3n}. \]
	\end{theorem}
	\begin{proof}
		Start with (\ref{cor:delta_ratio}) and set $n := 3n$. Apply (\ref{cor:A_shift}) and (\ref{def:A}). On the right side apply (\ref{cor:lifted_K_pure}) with $k := 6$. Then apply (\ref{cor:lifted_k_values}) on the right side to get the result.
	\end{proof}
	
	\begin{lemma}
		\label{lem:e2lift_two_variables}
		\begin{align*}
			-\lift{e_2}{n} &= x^{2n} + x^ny^n + y^{2n}\\
			&= y^{2n}+y^nz^n+z^{2n}\\
			&= z^{2n}+z^nx^n+x^{2n}
		\end{align*}
	\end{lemma}
	\begin{proof}
		It follows directly from (\ref{cor:lifted_e1e2}) using $s_n = 0$.
	\end{proof}
	
	\begin{lemma}
		\label{lem:gamma_3}
		\[ \lift{\Gdelta{3}}{n} = -\lift{e_2}{n}^3. \]
	\end{lemma}
	\begin{proof}
		It follows directly by multiplying all formulas from (\ref{lem:e2lift_two_variables}) and lifting (\ref{lem:gdelta3_two_variables}).
	\end{proof}
	
	\begin{theorem}
		\label{thm:A_identity}
		The scaling constant $A$ is given by
		\[ A = -\lift{e_2}{n}. \]
	\end{theorem}
	
	\begin{proof}
		Apply (\ref{thm:gamma_mult}) with $m := 3$ to get $\Gdelta{3n} = \lift{\Gdelta{3}}{n} \cdot \Gdelta{n}$. Substitute (\ref{lem:gamma_3}). Substituting into (\ref{thm:gamma_3n}):
		\[ \Gdelta{n} \cdot \lift{e_2}{n}^2 \cdot A = -\lift{e_2}{n}^3 \cdot \Gdelta{n}. \]
		Dividing by $\Gdelta{n} \cdot \lift{e_2}{n}^2$ yields $A = -\lift{e_2}{n}$.
	\end{proof}
	
	\begin{corollary}
		\label{cor:sn_zero_K}
		\[ 2e_1\,K(n+1) - e_2\,K(n) = 3K(n+2). \]
	\end{corollary}
	\begin{proof}
		Set $s_n = 0$ in (\ref{cor:sn_K}) and rearrange.
	\end{proof}
	
	\begin{lemma}
		\label{lem:K_3n2}
		\[ K(3n+2) = A \cdot K(n+2) + e_3^n. \]
	\end{lemma}
	\begin{proof}
		Apply (\ref{cor:sn_K}) at index $3n$. Substitute $s_{3n}$ from (\ref{cor:lifted_s_values}). Apply (\ref{def:A}) and (\ref{cor:A_shift}). Then apply (\ref{cor:sn_zero_K}). Dividing by 3 will get the result.
	\end{proof}
	
	\begin{theorem}
		\label{thm:master_identity}
		For any integer $j$:
		\[ K(3n+j) = A\cdot K(n+j) + e_3^n\cdot K(j) \]
	\end{theorem}
	\begin{proof}
		Both sides satisfy the same recurrence in $j$ (by (\ref{thm:universal_recurrence})), so it suffices to check equality at $j = 0, 1, 2$. For $j = 0$: both sides equal $K(3n)$ by (\ref{def:A}). For $j = 1$: both sides equal $K(3n+1)$ by (\ref{cor:A_shift}). For $j = 2$: this is (\ref{lem:K_3n2}), noting $K(2) = 1$ from (\ref{thm:initial_values}).
	\end{proof}
